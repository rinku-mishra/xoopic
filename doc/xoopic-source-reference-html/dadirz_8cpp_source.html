65 /**********************************************************************
66  Single Peaceman Rachford Douglas pass with Direchlet 0 c boundary
67  conditions for the equation:
68 
69  dtu = dxdxu + dydyu -s, where s is constant in time.
70 
71  The Crank-Nicolson finite difference approximation to the
72  above equation leads to the fractional step or
73  ADI equations:
74 
75  u*(i,j)-(del_t/2dxdx)[u*(i+1,j)-2u*(i,j)+u*(i-1,j)]
76  = un(i,j)+(del_t/2dydy)[un(i,j+1)-2un(i,j)+un(i,j-1)] - (del_t/2)s(i,j)
77 
78  un+1(i,j)-(del_t/2dydy)[un+1(i,j+1)-2un+1(i,j)+un+1(i,j-1)]
79  = u*(i,j)+(del_t/2dxdx)[u*(i+1,j)-2u*(i,j)+u*(i-1,j)] - (del_t/2)s(i,j)

85 void Dadirz::adi(Scalar **uadi, Scalar **s, Scalar del_t)
86 {
87  register int i, j;
88  Scalar dth;
89 
90  dth = .5*del_t;
91 
92  /***************************************/
93  /* Do z pass. Set up variables for */
94  /* tridiagonal inversion along z. */
95 
96  for(j=0;j<ng2;j++)
97  {
98  for(i=0; i<ng1; i++)
99  {
100  a_x1[i] = -dth*a_x1geom[i][j];
101  b_x1[i] = 1 - dth*b_x1geom[i][j];
102  c_x1[i] = -dth*c_x1geom[i][j];
103  if(j==0)
104  b_x1[i] -= 4*dth*b_x2geom[i][j];
105  }
106  /* handle the boundary conditions, neumann and dirichlet */
107 
108 
109  for(i=0; i<ng1; i++)
110  if(b_x2geom[i][j]!=0.0) // if this isn't dirichlet
111  r_x1[i] = uadi[i][j] + dth*(-s[i][j]
112  +((j>0)?(a_x2geom[i][j]*uadi[i][j-1]):0)
113  +((j!=0)?b_x2geom[i][j]*uadi[i][j]:0)
114  +((j<nc2)?
115  ((j==0)?4:1)
116  *c_x2geom[i][j]*uadi[i][j+1]
117  :0));
118  else
119  r_x1[i] = uadi[i][j]; // metal sets potential here
120 
121 
122  /* Solve tridiagonal system. */
123  tridag(a_x1, b_x1, c_x1, r_x1, v_x1, gam_x1, ng1);
124 
125 
126  /* Copy solution into ustar. */
127  for(i=0; i<ng1; i++) ustar[i][j] =v_x1[i];
128 
129  }
130 
131  /***************************************/
132  /* Do r pass. Set up variables for */
133  /* tridiagonal inversion along r. */
134 
135  for(i=0;i<ng1;i++) {
136  for(j=0; j<ng2; j++) {
137 
138  a_x2[j] = -dth*a_x2geom[i][j];
139  b_x2[j] = 1 - dth*b_x2geom[i][j];
140  c_x2[j] = -dth*c_x2geom[i][j];
141  // if(j==0&&N_Y0==2) {
142  if(j==0) {
143  b_x2[j] -= dth*(b_x1geom[i][j]+3*b_x2geom[i][j]);
144  c_x2[j] -= dth*(c_x2geom[i][j]*3);
145  }
146  }
147 
148 
149  for(j=0; j<ng2; j++)
150  if(b_x2geom[i][j]!=0) //non-dirichlet
151  r_x2[j] = ustar[i][j] +dth*(-s[i][j]
152  +((i>0)?a_x1geom[i][j]*ustar[i-1][j]:0)
153  +((j!=0)?b_x1geom[i][j]*ustar[i][j]:0)
154  +((i<nc1)?c_x1geom[i][j]*ustar[i+1][j]:0));
155  else
156  r_x2[j] = ustar[i][j]; // metal sets the potential here.
157 
158 
159  /* Solve tridiagonal system. */
160  tridag(a_x2, b_x2, c_x2, r_x2, v_x2, gam_x2, ng2);
161 
162  /* Copy solution into ustar. */
163  for(j=0; j<ng2; j++) uadi[i][j] =v_x2[j];
164  }
165 }

168 void Dadirz::init_solve(Grid *grid,Scalar **epsi)
169 {
170  register int i, j;
171  //set up freespace default coefficients everywhere.
172  //outside boundaries must be set up specially,
173  //but at this point we don't know these yet.
174  for(i=0;i<ng1;i++)
175  for(j=0;j<ng2;j++)
176  {
177  Boundary *B = grid->GetNodeBoundary()[i][j];
178  BCTypes type;
179  if(B!=NULL) type = B->getBCType();
180  else type = FREESPACE;
181  set_coefficient(i,j,type,grid);
182  }
183 
184  /************************************************************/
185  /* Determine initial step size such that 1=del_t/(dx1*dx1+dx2*dx2).
186  Chosen because this looks like stability condition for
187  an explicit scheme for the above parabolic equation.
188  Also calculate the usual constants. */
189 #ifndef sqr
190 #define sqr(x) ((x)*(x))
191 #endif
192 
193  Vector2 **X = grid->getX();
194  del_t0 = 0.1*(sqr(X[1][0].e1()-X[0][0].e1()) +sqr(X[0][1].e2()-X[0][0].e2()))/epsi[0][0];
195 
196  for(i=1; i<ng1; i++)
197  for(j=1;j<ng2;j++)
198  del_t0 = MIN(del_t0, 0.1*(sqr(X[i][j].e1()-X[i-1][j].e1())
199  +sqr(X[i][j].e2()-X[i][j-1].e2()))/epsi[0][0]);

207 void Dadirz::set_coefficient(int i, int j, BCTypes type, Grid *grid)
208 { Vector2 **X=grid->getX();
209  int I=grid->getJ();
210  int J=grid->getK();
211  // dz-
212  Scalar dxia = X[i][j].e1() -X[MAX(0,i-1)][j].e1();
213  // dz+
214  Scalar dxib = X[MIN(I,i+1)][j].e1()-X[i][j].e1();
215  // dz
216  Scalar dxave= 0.5*(dxia + dxib);
217 
218  Scalar ra = X[i][MAX(0,j-1)].e2();
219  Scalar r = X[i][j].e2();
220  Scalar rb = X[i][MIN(J,j+1)].e2();
221  Scalar ra2= 0.5*(ra + r);
222  Scalar rb2= 0.5*(r +rb);
223  Scalar dr2 = rb2*rb2-ra2*ra2;
224  Scalar drb = rb - r;
225  Scalar dra = r - ra;
226 
227 
228  Scalar da1a,da1b,da2aa,da2ab,da2ba,da2bb; // Areas of surfaces, for epsilons
229 
230  Scalar e1a,e1b,e2a,e2b; /* epsilon (i-1/2, j), (i+1/2,j),(i,j-1/2),(i,j+1/2) */
231 
232  // the area weightings to use for the epsilons
233  da1a = sqr(r) - sqr(ra2);
234  da1b = sqr(rb2)-sqr(r);
235  da2aa= dxia;
236  da2ab= dxib;
237  da2ba= dxia;
238  da2bb= dxib;
239 
240  // the epsilons to use for each coefficient below
241 // e1a = (epsi[i][j]*da1a + epsi[i][MIN(J,j+1)]*da1b ) / ( da1a + da1b);
242 // e1b = (epsi[MIN(I,i+1)][j]*da1a + epsi[MIN(I,i+1)][MIN(j+1,J)]*da1b ) / ( da1a + da1b);
243 // e2a = (epsi[i][j]*da2aa + epsi[MIN(I,i+1)][j] * da2ab ) / ( da2aa + da2ab);
244 // e2b = (epsi[i][MIN(J,j+1)]*da2ba + epsi[MIN(I,i+1)][j] * da2bb ) / ( da2aa + da2ab);
245  int im = MAX(i-1,0);
246  int jm = MAX(j-1,0);
247  e1a = (epsi[im][jm]*da1a + epsi[im][MIN(j,J-1)]*da1b ) / ( da1a + da1b);
248  e1b = (epsi[MIN(i,I-1)][jm]*da1a + epsi[MIN(i,I-1)][MIN(j,J-1)]*da1b ) / ( da1a + da1b);
249  e2a = (epsi[im][jm]*da2aa + epsi[MIN(i,I-1)][jm] * da2ab ) / ( da2aa + da2ab);
250  e2b = (epsi[im][MIN(j,J-1)]*da2ba + epsi[MIN(i,I-1)][MIN(j,J-1)] * da2bb ) / ( da2aa + da2ab);
251 
252  // If the dielectric boundary isn't at an edge, treat it like
253  // freespace
254  if(type==DIELECTRIC_BOUNDARY && (i!=0||i!=I||j!=J||j!=0)) type = FREESPACE;
255 
256 
257  switch(type)
258  {
259  case FREESPACE:
260  {
261  if(b_x2geom[i][j]==0) break; // if dirichlet, don't override.
262  //Z finite difference
263  if(i) a_x1geom[i][j] = e1a/(dxia*dxave);
264  if(i!=I) c_x1geom[i][j] = e1b/(dxib*dxave);
265  b_x1geom[i][j] = -( a_x1geom[i][j] + c_x1geom[i][j] );
266 
267  //R finite difference
268 
269  if(j) a_x2geom[i][j] = e2a*2*ra2/(dr2 * dra);
270  if(j!=J) c_x2geom[i][j] = e2b*2*rb2/(dr2 * drb);
271  b_x2geom[i][j] = - ( a_x2geom[i][j] + c_x2geom[i][j] );
272 
273  break;
274  }
275  case CONDUCTING_BOUNDARY:
276  {
277  a_x1geom[i][j]=0.0;
278  b_x1geom[i][j]=0.0;
279  c_x1geom[i][j]=0.0;
280  a_x2geom[i][j]=0.0;
281  b_x2geom[i][j]=0.0;
282  c_x2geom[i][j]=0.0;
283  break;
284  }
285  case DIELECTRIC_BOUNDARY:
286  {
287  if(i==0&&b_x2geom[i][j]!=0) { //Don't override a dirichlet with a Neumann
288  Scalar xa = X[i][j].e1();
289  Scalar xave = 0.5*(2*xa + dxib);
290 
291  a_x1geom[i][j] = 0.0;
292  c_x1geom[i][j] = e1b/(xave-xa)/dxib;
293  b_x1geom[i][j] = -( a_x1geom[i][j] + c_x1geom[i][j] );
294  if(j==0) {
295  Scalar dria = X[i][1].e2() - X[i][0].e2();
296  a_x2geom[i][0] = 0.0;
297  b_x2geom[i][0] = -e2b/dria/dria;
298  c_x2geom[i][0] = -b_x2geom[i][0];
299  }
300  if(j==J) { //neumann neumann corner
301  c_x2geom[i][j] = 0.0;
302  a_x2geom[i][j] = e2a*2*ra2/(dr2*dra);
303  b_x2geom[i][j] = -a_x2geom[i][j];
304  }
305 
306  }
307  if(i==I&&b_x2geom[i][j]!=0) {
308  Scalar xb = X[i][j].e1();
309  Scalar xave = 0.5*(X[i][j].e1() + X[i-1][j].e1());
310 
311  a_x1geom[i][j] = e1a/(xb-xave)/dxia;
312  c_x1geom[i][j] = 0.0;
313  b_x1geom[i][j] = -( a_x1geom[i][j] + c_x1geom[i][j] );
314  if(j==0&&b_x2geom[i][j]!=0) { //neumann neumann corner
315  Scalar dria = X[i][j+1].e2() - X[i][j].e2();
316  a_x2geom[i][j] = 0.0;
317  c_x2geom[i][j] = e2b/dria/dria;
318  b_x2geom[i][j] = -c_x2geom[i][j];
319  }
320  if(j==J&&b_x2geom[i][j]!=0) { //neumann neumann corner
321  c_x2geom[i][j] = 0.0;
322  a_x2geom[i][j] = e2a*2*ra2/(dr2*dra);
323  b_x2geom[i][j] = -a_x2geom[i][j];
324  }
325  }
326 
327  if(j==J && i!=0&&b_x2geom[i][j]!=0 ) {
328  Scalar dria = X[i][j].e2()-X[i][j-1].e2();
329 
330  a_x2geom[i][nc2] = 2 * e2a * ra2/ ( dria * ( r*r - ra2*ra2));
331  b_x2geom[i][nc2] = -a_x2geom[i][nc2];
332  c_x2geom[i][nc2] = 0.0;
333  }
334  break;
335  }
336  case CYLINDRICAL_AXIS:
337  { if(i==0||i==I) break; // don't want cyl_axis in corners
338  Scalar dria = X[i][1].e2() - X[i][0].e2();
339  a_x2geom[i][0] = 0.0;
340  b_x2geom[i][0] = -e2b/dria/dria;
341  c_x2geom[i][0] = -b_x2geom[i][0];
342 
343  a_x1geom[i][j] = e1a/(dxia*dxave);
344  c_x1geom[i][j] = e1b/(dxib*dxave);
345  b_x1geom[i][j] = -( a_x1geom[i][j] + c_x1geom[i][j] );
346 
347  break;
348  }
349  default:
350  {
351  cout << "DADI doesn't know about this boundary condition type!\n;" << endl;
352  cout << "(was either PERIODIC_BOUNDARY or SPATIAL_REGION_BOUNDARY)\n" << endl;
353  //e_xit(1);
354  }
355  }
356  }

364 int Dadirz::solve(Scalar **u_in, Scalar **s, int itermax,Scalar tol_test)
365 {
366  register int i, j;
367  int iter, ndiscard;
368  static Scalar del_t=0.0;
369  Scalar del_td=0, tptop=0, tpbot=0, ratio=0;
370  Scalar rnorm=0, rsum=0, res=0, errchk=0, dxdxutrm=0, dydyutrm=0;
371 
372  rnorm=rsum=0.0;
373  for(i=0; i< ng1; i++)
374  for(j=0; j< ng2; j++) {
375 
376  /* Residual normalization. */
377  // dirichlet conditions don't add to rnorm
378  rnorm +=( (b_x2geom[i][j]==0)? 0:s[i][j]*s[i][j]) ;
379 
380  /* copy u_in to u for working purposes. */
381  u[i][j]=(Scalar)u_in[i][j];
382 
383  //calculate an initial estimate of the residual
384  /* Use the residual as the absolute error and if it is bigger
385  than the stored maximum absolute error, record new maximum
386  absolute error and location. */
387 
388  if(i>0&&j>0&&i<nc1&&j<nc2) {
389  dxdxutrm=a_x1geom[i][j]*u_in[i-1][j] +b_x1geom[i][j]*u_in[i][j] +
390  c_x1geom[i][j]*u_in[i+1][j];
391  dydyutrm=a_x2geom[i][j]*u_in[i][j-1] +b_x2geom[i][j]*u_in[i][j] +
392  c_x2geom[i][j]*u_in[i][j+1];
393  }
394  /* only include points which are not in structures. */
395  errchk = ((b_x2geom[i][j]==0)? 0:dxdxutrm +dydyutrm -s[i][j]);
396 
397  /* Residual sums. */
398  rsum += errchk*errchk;
399  }
400 
401  // If rnorm is zero, we must deal with it...
402  if (rnorm == 0.0) {
403 
404  // check dirichlet conditions
405  for(i=0;i<ng1;i++) for(j=0;j<ng2;j++) {
406  // check left
407  rnorm += sqr(((i>0&&b_x2geom[i-1][j]!=0)?c_x1geom[i-1][j]*u[i][j]:0));
408  // check right
409  rnorm += sqr(((i<nc1&&b_x2geom[i+1][j]!=0)?a_x1geom[i+1][j]*u[i][j]:0));
410  // check up
411  rnorm += sqr(((j>0&&b_x2geom[i][j-1]!=0)?c_x2geom[i][j-1]*u[i][j]:0));
412  // check right
413  rnorm += sqr(((j<nc2&&b_x2geom[i][j+1]!=0)?c_x2geom[i][j+1]*u[i][j]:0));
414 
415  }
416  //rnorm = sqrt(rnorm);
417  if(rnorm == 0) { //still zero, we don't need to iterate
418  for(i=0;i<ng1;i++) for(j=0;j<ng2;j++) u_in[i][j]=0;
419  return 0;
420  }
421  }
422  rnorm=sqrt(rnorm);
423  res = sqrt(rsum)/rnorm;
424 #ifdef DADI_DEBUG
425  printf("dadi: res = %g\n",res);
426  printf("dadi: rnorm= %lg\n", rnorm);
427 #endif
428 
429  if(res<tol_test) return 0; // no need to iterate
430 
431  /*************************************************/
432  if(del_t<1.0e-10*del_t0) del_t=del_t0; else del_t/=4;
433  del_td= 2.0*del_t;
434  ndiscard=0;
435 
436  /********************/
437  /* Begin iteration. */
438 
439  for(iter=0; iter<itermax; iter++)
440  {
441  /*************************************************/
442  /* Copy u into the work array and storage array. */
443 
444  for(i=0; i< ng1; i++)
445  for(j=0; j< ng2; j++) uwork[i][j] = ustor[i][j] = u[i][j];
446 
447  /************************************/
448  /* Two advances of u via ADI at del_t. */
449 
450  adi(u, s, del_t);
451  adi(u, s, del_t);
452 
453  /*****************************************/
454  /* One advance of uwork via ADI at 2*del_t. */
455 
456  adi(uwork, s, del_td);
457 
458  /*******************************************************/
459  /* Calculate test parameter and normalized error.
460  For Dirichlet BCs, no need to worry about boundary
461  points since u,uwork, and ustor should be the same. */
462 
463  tptop= tpbot= rsum = 0.0;
464 
465  for(i=1; i< nc1; i++)
466  for(j=1; j< nc2; j++)
467  {
468  /* Test paramter sums. */
469  tptop += (u[i][j]-uwork[i][j])*(u[i][j]-uwork[i][j]);
470  tpbot += (u[i][j]-ustor[i][j])*(u[i][j]-ustor[i][j]);
471 
472  /* Residual terms. */
473 
474  dxdxutrm=a_x1geom[i][j]*u[i-1][j] +b_x1geom[i][j]*u[i][j] +c_x1geom[i][j]*u[i+1][j];
475  dydyutrm=a_x2geom[i][j]*u[i][j-1] +b_x2geom[i][j]*u[i][j] +c_x2geom[i][j]*u[i][j+1];
476 
477  /* Use the residual as the absolute error and if it is bigger
478  than the stored maximum absolute error, record new maximum
479  absolute error and location. */
480  /* only include points which are outside of structures. */
481  errchk = ((b_x2geom[i][j]==0)? 0:dxdxutrm +dydyutrm -s[i][j]);
482 
483  /* Residual sums. */
484  rsum += errchk*errchk;
485  }
486 
487  /* Calculate normalized residual. */
488  res = sqrt(rsum)/rnorm;
489 #ifdef DADI_DEBUG
490  printf("dadi: iter= %d, res = %lg del_t=%lf\n", iter, res,del_t);
491 #endif
492  /* If the residual is less than the tolerance, SUCCESS! */
493  if ((res < tol_test) && (iter))
494  {
495 #ifdef DADI_DEBUG
496  printf("dadi: iter=%d\n", iter);
497 #endif
498  for(i=0;i<ng1;i++)
499  for(j=0;j<ng2;j++)
500  u_in[i][j]=(Scalar)u[i][j];
501 
502  return(0);
503  }
504 
505  /* Determine ratio used to find the time step change. If tpbot
506  is zero but tptop is finite, consider this a case of a large
507  ratio and act accordingly. DWH does about the same thing
508  except he does NOT discard the solution if tpbot=0. */
