53 void PortTM::applyFields(Scalar t,Scalar dt)
54 {
55  if(dt==0) return;
56  // this could be cleaned up by using a funciton pointer sin for XY and bessj1 for RX
57  // also gamma1 and M_PI would have to be selected
58 // cout << "dt " << dt << endl;
59 // cout << "time " << t << endl;
60  time = t- dt/2;
61  grid = fields->get_grid();
62 // EnergyOut = 0.0;
63  if (fields->getSub_Cycle_Iter() ==1) EnergyOut = 0.0;
64  else if (fields->getSub_Cycle_Iter() ==0) EnergyOut = 0.0;
65  A1 = get_time_value(t);
66 // cout << "A1 " << A1 << endl;
67  if (grid->query_geometry() == ZRGEOM)
68  {
69  a = grid->getMKS(0, k2).e2();
70  Scalar max_bessj1 = bessj1(1.841); //scaling bessj1 by it's peak
71  for (int k = k1; k < k2; k++)
72  {
73  rforEr = (grid->getMKS(j1, k + 1).e2() +
74  grid->getMKS(j1, k).e2())/2.0;
75  Er = A1*bessj1(gamma0*rforEr/a)/max_bessj1;
76  intEr = grid->dl2(j1, k)*Er;
77  fields->setIntEdl(j1, k, 2, intEr);
78  if(EFFlag)
79  {
80  if (fields->getiC()[j2+normal][k].e2())
81  {
82  // Er = get_time_value(time)*bessj1(gamma0*rforEr/a)/max_bessj1;
83  HonBoundary = (fields->getiL()[j2+shift][k].e3())*
84  (fields->getIntBdS()[j2+shift][k].e3()) +
85  normal*.5*(fields->getIMesh(j2,k).e2() +
86  get_time_value_deriv(time)/(fields->getiC()[j2+normal][k].e2()));
87  EnergyOut += normal*Er*HonBoundary*grid->dS(j2+normal,k).e1();
88  }
89  }
90  }
91  }
92  if (grid->query_geometry() == ZXGEOM)
93  {
94  if (alongx2())
95  {
96  a = (grid->getMKS(j2,k2) - grid->getMKS(j1,k1)).e2();
97  for (int k = k1; k < k2; k++)
98  {
99  Scalar yforEy = (grid->getMKS(j1, k + 1).e2() +
100  grid->getMKS(j1, k).e2())/2.0 - grid->getMKS(j1,k1).e2() ;
101  Scalar Ey = A1*cos(M_PI*yforEy/a);
102  Scalar intEy = grid->dl2(j1, k)*Ey;
103  fields->setIntEdl(j1, k, 2, intEy);
104  if(EFFlag)
105  {
106  if (fields->getiC()[j2+normal][k].e2())
107  {
108  HonBoundary = (fields->getiL()[j2+shift][k].e3())*
109  (fields->getIntBdS()[j2+shift][k].e3()) +
110  normal*.5*(fields->getIMesh(j2,k).e2() +
111  get_time_value_deriv(time)/(fields->getiC()[j2+normal][k].e2()));
112  EnergyOut += normal*Ey*HonBoundary*grid->dS(j2+shift,k).e1();
113  }
114  }
115  }
116  }
117  else
118  {
119  a = (grid->getMKS(j2,k2) - grid->getMKS(j1,k1)).e1();
120  for (int j = j1; j < j2; j++)
121  {
122  Scalar xforEx = (grid->getMKS(j, k1).e1() +
123  grid->getMKS(j+1, k1).e1())/2.0 - grid->getMKS(j1,k1).e1();
124  Scalar Ex = A1*cos(M_PI*xforEx/a);
125  Scalar intEx = grid->dl1(j, k1)*Ex;
126  fields->setIntEdl(j, k1, 1, intEx);
127  if(EFFlag)
128  {
129  if (fields->getiC()[j][k1+normal].e1())
130  {
131  HonBoundary = (fields->getiL()[j][k1+shift].e3())*
132  (fields->getIntBdS()[j][k1+shift].e3()) +
133  normal*.5*(fields->getIMesh(j,k1).e2() +
134  get_time_value_deriv(time)/(fields->getiC()[j][k1+normal].e1()));
135  EnergyOut += normal*Ex*HonBoundary*grid->dS(j,k1+normal).e2();
136  }
137  }
138  }
139  }
140  }
141  if (fields->getFieldSub() == fields->getSub_Cycle_Iter()) EnergyOut /= (fields->getFieldSub());
142 }
