65 void PlasmaSource::initialize()
66 {
67  init = FALSE;
68  Grid *grid = fields->get_grid();
69  if (grid->query_geometry() == ZXGEOM) is_xy = TRUE;
70  else is_xy = FALSE;
71  p1 = fields->getMKS(j1, k1);
72  p2 = fields->getMKS(j2, k2);
73  p1 += 1E-6*p2;// move off the edge a small amount
74  p2 -= 1E-6*p2;
75  deltaP = p2-p1;
76  Scalar volume = deltaP.e1();
77  if (is_xy) volume *= deltaP.e2();
78  else
79  {
80  rMinSqr = p1.e2()*p1.e2();
81  drSqr = p2.e2()*p2.e2()-rMinSqr;
82  volume *= M_PI*drSqr;
83  }
84  sourceRate *= volume/np2c; // convert to num/time
85 
86  // Markov Chain Monte Carlo Load
87  // doing a one time burn in
88  int BurnNumber = 10000;
89  Vector2 xc;
90  // initial guess of x0
91  x0.set_e1(p1.e1()+deltaP.e1()*frand());
92  if(is_xy)
93  x0.set_e2(p1.e2()+deltaP.e2()*frand());
94  else
95  x0.set_e2(sqrt(drSqr*frand()+p1.e2()));
96  int seed = 0;
97 
98  // burn in
99  Fx0 = F(x0,0);
100  while (seed < BurnNumber)
101  {
102  xc.set_e1(p1.e1()+deltaP.e1()*frand());
103  if(is_xy)
104  xc.set_e2(p1.e2()+deltaP.e2()*frand());
105  else
106  xc.set_e2(sqrt(drSqr*frand()+rMinSqr));
107  Fxc = F(xc,0);
108  if ((Fxc!=0)&&((Fx0==0)||(Fxc>=Fx0*(Scalar)frand())))
109  {
110  seed++;
111  x0=xc;
112  Fx0=Fxc;
113  }
114  }
115 
116  return;
117 }
