57 void Gap::applyFields(Scalar t,Scalar dt)
58 {
59  if(dt==0) return;
60  time = t - dt/2;
61 // if ((fields->getSub_Cycle_Iter) == 1) EnergyOut = 0.0;
62 // else if ((fields->getSub_Cycle_Iter) ==0) EnergyOut = 0.0;
63  if (fields->getSub_Cycle_Iter() ==1) EnergyOut = 0.0;
64  else if (fields->getSub_Cycle_Iter() ==0) EnergyOut = 0.0;
65  grid = fields->get_grid();
66 
67  Scalar newE = get_time_value(t);
68  Scalar newEdl;
69  if (alongx1())
70  for (int j=MIN(j1,j2); j<MAX(j1,j2); j++)
71  {
72  newEdl = newE*grid->dl1(j, k1);
73  fields->setIntEdl(j, k1, 1, newEdl);
74  if(EFFlag)
75  {
76  if (fields->getiC()[j][k1+normal].e1())
77  {
78  HonBoundary = fields->getiL()[j][k1+shift].e3()*
79  fields->getIntBdS()[j][k1+shift].e3()+
80  normal*.5*(fields->getIMesh(j,k1).e1()+
81  get_time_value_deriv(time)/fields->getiC()[j][k1+normal].e1());
82  EnergyOut += normal*get_time_value(time)*HonBoundary*grid->dS(j,k1+shift).e2()/(grid->dl(j,k2+shift).e3());
83  }
84  }
85  }
86  else
87  for (int k=MIN(k1, k2); k<MAX(k1, k2); k++)
88  {
89  newEdl = newE*grid->dl2(j1, k);
90  fields->setIntEdl(j1, k, 2, newEdl);
91  if(EFFlag)
92  {
93  if (fields->getiC()[j1+normal][k].e2())
94  {
95  HonBoundary = fields->getiL()[j1+shift][k].e3()*
96  fields->getIntBdS()[j1+shift][k].e3() +
97  normal*.5*(fields->getIMesh(j1,k).e2() +
98  get_time_value_deriv(time)/fields->getiC()[j1+normal][k].e2());
99  Scalar g = grid->dl2(j2+shift,k);
100  if (g!=0)
101  EnergyOut += normal*get_time_value(time)*HonBoundary*grid->dS(j1+shift,k).e1()/g;
102  }
103  }
104  }
105 }
