251  This function adds point style data to the plot. Values \a xval and

257 
258 void TxqPlot::AppendPointData(double xval, double yval)
259 {
260 
261  // Add to the next element of the vector
262  x_.Append( xval );
263  y_.Append( yval );
264  next++;
265 
266  // If autoRanging is turned on, check if mins/maxes should be adjusted
267  if (xAutoRange) {
268  if (xval > xmax) xmax = xval;
269  if (xval < xmin) xmin = xval;
270  }
271  if (yAutoRange) {
272  if (yval > ymax) ymax = yval;
273  if (yval < ymin) ymin = yval;
274  }
275 
276  // Message out the slot that we have updated.
277  emit new_data();
278  update();
279 }
