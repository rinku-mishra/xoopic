117 // initialPush() pushes emitted particles through a smaller timestep,
118 // del_t. dt specifies the full timestep for this species, x is the initial
119 // position, u is the initial momentum. Note that u = gamma*v. The
120 // position is advanced using the initial velocity, 1st order accurate.
121 // The u0 and other 0 variable series correspond to a 1st order half push for
122 // the velocity, which will be used to obtain x. Refer to the better than
123 // 1st order method in Cartwright et. al. Injection ...
124 
125 Boundary* Emitter::initialPush(Scalar del_t, Scalar dt, Particle &p)
126 {
127  Vector3 u = p.get_u();
128  Vector2 x = p.get_x();
129  Vector3 u0 = u;
130  Scalar igamma = 1/p.gamma();
131  Scalar f = 0.5*(del_t - 0.5*dt)*p.get_q_over_m();
132  Scalar f0 = 0.25*del_t*p.get_q_over_m();
133  Vector3 a = f*fields->E(x);
134  Vector3 a0 = f0*fields->E(x);
135  u += a;
136  u0 += a0;
137  Vector3 tt = fields->B(x);
138  if (tt.e1()!=0 || tt.e2()!=0 || tt.e3()!=0){ // magnetized push
139  Vector3 tt0 = tt*f0*igamma;
140  tt *= f*igamma;
141  Vector3 u1 = u + u.cross(tt);
142  Vector3 u2 = u0 + u0.cross(tt0);
143  Vector3 ss = 2*tt/(1 + tt*tt);
144  Vector3 ss0 = 2*tt0/(1 + tt0*tt0);
145  u += u1.cross(ss);
146  u0 += u2.cross(ss0);
147  }
148  u += a;
149  u0 += a0;
150  Vector3 dxMKS = fields->differentialMove(fields->getMKS(x), u0*igamma, u0, del_t);
151  Boundary* bPtr = fields->translateAccumulate(x, dxMKS, p.get_q()/dt);
152  p.set_u(u);
153  p.set_x(x);
154  if (bPtr) bPtr->collect(p, dxMKS);
155  else{
156  if (igamma>.86)
157  species->addKineticEnergy(.5*u0*igamma*u*p.get_m());
158  else
159  species->addKineticEnergy(p.get_m()*(p.gamma()-1)*SPEED_OF_LIGHT_SQ);
160  }
161  return bPtr;
162 }
