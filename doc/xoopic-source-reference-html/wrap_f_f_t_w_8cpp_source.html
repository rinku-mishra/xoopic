231 void WrapFFTW::map2DBackwardRFFTW(Scalar** reOut, Scalar** imOut) {
232  int N1 = n[0];
233  int N2 = n[1];
234  int j, k, index;
235  int N2over2 = N2/2;
236  int N2over2plus1 = N2over2+1;
237  for ( j = 0; j < N1; j++ ) {
238  for ( k = 0; k < N2; k++ ) {
239  if ( k > N2over2 ) {
240  //
241  // use symmetry and periodicity to assign in this case
242  //
243  if ( j == 0 )
244  index = N2-k;
245  else
246  index = (N1-j)*N2over2plus1+(N2-k);
247  reOut[j][k] = cDataOut[index].re;
248  imOut[j][k] = -cDataOut[index].im;
249  } else {
250  index = j*N2over2plus1+k;
251  reOut[j][k] = cDataOut[index].re;
252  imOut[j][k] = cDataOut[index].im;
253  }
