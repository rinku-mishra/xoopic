133 void Dielectric::collect(Particle& p, Vector3& dxMKS)
134 {
135  int j=0;
136  Scalar w=0.;
137  if ((Scalar)frand() <= reflection && p.energy_eV_MKS() <= refMaxE){ // reflect particles
138  Vector3 u = p.get_u();
139  Vector2 x = p.get_x();
140  if (alongx2()){
141  u.set_e1(-u.e1()); // reverse u comp. normal to bc
142  dxMKS.set_e1(-dxMKS.e1()); // reverse the movement
143  if (u.e1() < 0) x.set_e1(x.e1()*.99999);
144  else x.set_e1(x.e1()*1.00001);
145  if (x.e2()==(int)x.e2()) x.set_e2(x.e2()*1.00001); // move off corners
146  else if (x.e2()==(int)x.e2()+1) x.set_e2(x.e2()*0.99999);
147  }
148  else {
149  u.set_e2(-u.e2());
150  dxMKS.set_e2(-dxMKS.e2()); // reverse the movement
151  if (u.e2() < 0) x.set_e2(x.e2()*.99999);
152  else x.set_e2(x.e2()*1.00001);
153  if (x.e1()==(int)x.e1()) x.set_e1(x.e1()*1.00001); // move off corners
154  else if (x.e1()==(int)x.e1()+1) x.set_e1(x.e1()*0.99999);
155  }
156  // note that the dt used below should probably be the remaining fraction for the
157  // move rather than the full dt.
158  Boundary *bPtr = fields->translateAccumulate(x, dxMKS, p.get_q()/fields->get_dt());
159  p.set_u(u);
160  p.set_x(x);
161  if (bPtr) bPtr->collect(p, dxMKS);
162  else particleList.add(&p);
163  return;
164  }
165  else if ((Scalar)frand() < transparency) {
166  // note that the dt used below should probably be the remaining fraction for the
167  // move rather than the full dt.
168  Vector2 x = p.get_x();
169  Boundary *bPtr = fields->translateAccumulate(x, dxMKS, p.get_q()/fields->get_dt());
170  p.set_x(x);
171  if (bPtr) bPtr->collect(p, dxMKS);
172  else particleList.add(&p);
173  return;
174  }
175  if (QuseFlag) {
176  if (alongx2()){ // vertical
177  j = (int) p.get_x().e2();
178  w = p.get_x().e2() - j;
179  j -= k1;
180  }
181  else {
182  j = (int) p.get_x().e1();
183  w = p.get_x().e1() - j;
184  j -= j1;
185  }
186  Q[j] += p.get_q()*(1-w);
187  Q[MIN(j+1, nPoints-1)] += p.get_q()*w;
188  }
189  Boundary::collect(p, dxMKS); // do statistics;
190  if (secondaryList.nItems()){
191  p.set_x(p.get_x() + _delta); // perturb off edge;
192  oopicListIter<Secondary> sIter(secondaryList);
193  for (; !sIter.Done(); sIter++){
194  int n = sIter()->generateSecondaries(p, particleList);
195  if (QuseFlag){
196  Scalar q = n*p.get_np2c()*sIter()->get_q();
197  Q[j] -= q*(1-w);
198  Q[MIN(j+1, nPoints-1)] -= q*w;
199  }
200  }
201  }
202  delete &p;
203 }

625 void DielectricRegion::collect(Particle& p, Vector3& dxMKS)
626 {
627  int j=0;
628  EDGE _edge;
629  Scalar xIntercept=0.;
630  // Compute the edge which collected:
631  if (fabs(p.get_x().e1() - j1) < DELTA) // left edge
632  {
633  xIntercept = p.get_x().e2();
634  _edge = LEFT;
635  _delta.set_e1(j1 ? -DELTA*j1 : -DELTA);
636  _delta.set_e2(0);
637  _unitNormal.set_e1(-1);
638  _unitNormal.set_e2(0);
639  }
640  else if (fabs(p.get_x().e1() - j2) < DELTA) // right edge
641  {
642  xIntercept = p.get_x().e2();
643  _edge = RIGHT;
644  _delta.set_e1(DELTA*j2);
645  _delta.set_e2(0);
646  _unitNormal.set_e1(1);
647  _unitNormal.set_e2(0);
648  }
649  else if (fabs(p.get_x().e2() - k1) < DELTA) // bottom edge
650  {
651  xIntercept = p.get_x().e1();
652  _edge = BOTTOM;
653  _delta.set_e1(0);
654  _delta.set_e2(k1 ? -DELTA*k1 : DELTA);
655  _unitNormal.set_e1(0);
656  _unitNormal.set_e2(-1);
657  }
658  else if (fabs(p.get_x().e2() - k2) < DELTA) // top edge
659  {
660  xIntercept = p.get_x().e1();
661  _edge = TOP;
662  _delta.set_e1(0);
663  _delta.set_e2(DELTA*k2);
664  _unitNormal.set_e1(0);
665  _unitNormal.set_e2(1);
666  }
667  else {
668  cout << "DielectricRegion::collect failed to find an edge" << endl;
669  printf("delta=(%g,%g) x=(%g,%g) s=%s j=%d, k=%d\n",
670  _delta.e1(), _delta.e2(), p.get_x().e1(), p.get_x().e2(),
671  p.get_speciesPtr()->get_name().c_str(), j1, k1);
672 // JRC: if no edge, cannot continue.
673  return;
674  }
675  if (QuseFlag)
676  {
677  j = (int) xIntercept;
678  addQ(_edge, j, xIntercept - j, p.get_q());
679  }
680  Boundary::collect(p, dxMKS); // for stats
681  if (secondaryList.nItems()){
682  p.set_x(p.get_x() + _delta); // perturb off edge;
683  oopicListIter<Secondary> sIter(secondaryList);
684  for (; !sIter.Done(); sIter++){
685  int n = sIter()->generateSecondaries(p, particleList);
686  if (QuseFlag && n>0){
687  Scalar q = -n*p.get_np2c()*sIter()->get_q();
688  addQ(_edge, j, xIntercept - j, q);
689  }
690  }
691  }
692  delete &p;
693 }

706 // Add q to the surface charge array; must be mapped from perimeter
707 // to linear array.
708 void DielectricRegion::addQ(EDGE edge, int i, Scalar w, Scalar q)
709 {
710  int index1=0;
711  int index2=0;
712  switch (edge)
713  {
714  case BOTTOM:
715  index1 = i - j1;
716  index2 = index1 + 1;
717  break;
718  case RIGHT:
719  index1 = i - k1 + j2 - j1;
720  index2 = index1 + 1;
721  break;
722  case TOP:
723  index1 = 2*j2 - j1 + k2 - k1 - i;
724  index2 = index1 - 1;
725  break;
726  case LEFT:
727  index1 = 2*(j2 - j1) + 2*k2 - k1 - i;
728  index2 = index1 - 1;
729  index1 %= nPoints; // wrap around to 0
730  break;
731  }
732  Q[index1] += q*(1 - w);
733  Q[index2] += q*w;
734 }
