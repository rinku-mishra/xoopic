329 void TxqPlotWindow::getYTicks( TxqList<double> &ticks )
330 {
331  double ymin = plot->Ymin();
332  double ymax = plot->Ymax();
333  double pos;
334  double diff = fabs( ymax - ymin );
335  double ld = log10( diff );
336  int scale =(int) ceil( ld );
337 
338  double normalizedDiff = pow( 10., ld - scale ); // .1 -> 1
339 
340  // From the normalized difference,
341  //determine what gap to place tick marks at
342  double anti = 2.; // Average number of tick intervals
343  double ts;
344  if (normalizedDiff/anti >= .05) ts = .05;
345  if (normalizedDiff/anti >= .1) ts = .1;
346  if (normalizedDiff/anti >= .2) ts = .2;
347  if (normalizedDiff/anti >= .4) ts = .4;
348 
349  double gap = ts * pow( 10., scale );
350  // double gap = 1.0 * pow( 10., scale );
351 
352 
353  double pos1 =gap*ceil(ymin/gap);
354  double pos2 =gap*ceil(ymax/gap);
355  pos=(((pos2-pos1)/2.0));
356 
357  ticks.Append(pos1);
358  ticks.Append(pos);
359  ticks.Append(pos2);
