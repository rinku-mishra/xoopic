877 /* Method for calculating energy imparted to a secondary (ejected)
878  * electron, using a parametric model that captures relativistic
879  * physics for large impact energy and agrees with data at low energy.
880  *
881  * The calculations underlying this method are contained in the
882  * Tech-X notes, "RNG Calculations for Scattering in XOOPIC," by
883  * David Bruhwiler (August 28, 2000).
884  *
885  * The parametric model is a relativistic generalization of the
886  * work by Rudd et al., Phys. Rev. A 44, 1644 (1991) and Phys.
887  * Rev. A 47, 1866 (1993). The relativistic generalization is
888  * described in the Tech-X notes, "Rudd's Differential Cross
889  * Section for Electron Impact Ionization," by David Bruhwiler,
890  * (July 17, 2000).
891  */
892 Scalar MCCPackage::ejectedEnergy(const CrossSection& cx,
893  const Scalar& impactEnergy) {
894 
895  // First, we need to calculate a random # between 0 and A_>,
896  // where A_> = f1(T) * (T-I) / f2(T) / (T+I), with I the threshold
897  // ionization energy and T the kinetic energy of the impacting
898  // electron "impactEnergy".
899 
900  Scalar I = cx.get_threshold();
901  if (impactEnergy <= I) return 0.;
902 
903  Scalar tPlusMC = impactEnergy + ELECTRON_MASS_EV;
904  Scalar twoTplusMC = impactEnergy + tPlusMC;
905  Scalar tPlusI = impactEnergy + I;
906  Scalar tMinusI = impactEnergy - I;
907 
908  Scalar invTplusMCsq = 1. / ( tPlusMC * tPlusMC );
909  Scalar tPlusIsq = tPlusI * tPlusI;
910  Scalar iOverT = I / impactEnergy;
911 
912 
913  Scalar funcT1 = 14./3. + .25 * tPlusIsq * invTplusMCsq
914  - ELECTRON_MASS_EV * twoTplusMC * iOverT * invTplusMCsq;
915 
916  Scalar funcT2 = 5./3. - iOverT - 2.*iOverT*iOverT/3.
917  + .5 * I * tMinusI * invTplusMCsq
918  + ELECTRON_MASS_EV*twoTplusMC*I*invTplusMCsq*log(iOverT)/tPlusI;
919 
920  Scalar aGreaterThan = funcT1 * tMinusI / funcT2 / tPlusI;
921 
922  // We may have to try several times before the "rejection method"
923  // lets us keep our ejected energy test value, wTest.
924  // Here we define a bool for the while construct and also
925  // declare some floating point variables needed inside.
926 
927  int needToTryAgain = 1;
928  Scalar randomFW, wTest, wPlusI, wPlusIsq;
929  Scalar invTminusW, invTminusWsq, invTminusW3;
930  Scalar probabilityRatio;
931 
932  while (needToTryAgain==1) {
933 
934  // The random number, called F(W) in the notes and randomFW here,
935  // is the antiderivative of a phony_but_simple probability, which
936  // is always >= the correct_but_messy probability.
937 
938  // do {randomFW=frand();} while (randomFW<=0. || randomFW>=1.);
939  randomFW = frand() * aGreaterThan;
940 
941  // wTest is the inverse of F(W)
942  wTest = I*funcT2*randomFW/(funcT1-funcT2*randomFW);
943 
944  // Because we are not working directly with the distribution function,
945  // we must use the "rejection method". This involves generating
946  // another random number and seeing whether it is > the ratio of
947  // the true probability over the phony_but_simple probability.
948 
949  wPlusI = wTest + I;
950  wPlusIsq = wPlusI * wPlusI;
951  invTminusW = 1./(impactEnergy-wTest);
952  invTminusWsq = invTminusW * invTminusW;
953  invTminusW3 = invTminusW * invTminusWsq;
954 
955  probabilityRatio = ( 1. + 4.*I/wPlusI/3. + wPlusIsq*invTminusWsq
956  + 4.*I*wPlusIsq*invTminusW3/3.
957  - ELECTRON_MASS_EV*twoTplusMC*wPlusI*invTminusW*invTplusMCsq
958  + wPlusIsq*invTplusMCsq ) / funcT1;
959 
960  if ( probabilityRatio >= (Scalar)frand() ) needToTryAgain = 0;
961  // else cout << "** wTest was rejected..." << endl;
962  } // end of while construct
963  // cout << "** wTest was accepted!" << endl << endl;
964 
965  // wTest is the energy of the ejected particle; the scattered primary
966  // has energy T-(W+I)
967 
968  // cout << endl;
969  // cout << " impact energy = " << impactEnergy << endl;
970  // cout << " ejected energy = " << wTest << endl;
971  return wTest;
972 }

976  * respect to the initial direction of the primary electron, for
977  * both the primary and secondary electrons in an impact-ionizaiton
978  * event.
979  *
980  * Unchanged imput variables are eImpact and eSecondary.
981  * Empty variables passed in by reference to hold the desired results
982  * are: thetaPrimary, thetaSecondary, phiPrimary and phiSecondary.
983  *
984  * The calculations underlying this method are contained in the
985  * Tech-X notes, "RNG Calculations for Scattering in XOOPIC," by
986  * David Bruhwiler (August 28, 2000).
987  *
988  * The parametric model is discussed in the Tech-X notes, "Rudd's
989  * Differential Cross Section for Electron Impact Ionization,"
990  * by David Bruhwiler, (July 17, 2000).
991  */
992 void MCCPackage::primarySecondaryAngles(const CrossSection& cx,
993  const Scalar& eImpact, const Scalar& eSecondary,
994  Scalar& thetaPrimary, Scalar& phiPrimary,
995  Scalar& thetaSecondary, Scalar& phiSecondary) {
996 
997  // Bail out, if impact energy < ionization threshold energy.
998 
999  Scalar I = cx.get_threshold();
1000  if (eImpact <= I) {
1001  thetaPrimary = 0.;
1002  thetaSecondary = 0.;
1003 
1004  phiPrimary = 0.;
1005  phiSecondary = 0.;
1006 
1007  return;
1008  }
1009 
1010  // First, we generate a Monte Carlo value for thetaPrimary --
1011  // We begin with the definition of some temporaries:
1012 
1013  Scalar ePrimary = eImpact - eSecondary - I;
1014 
1015  Scalar w = ePrimary;
1016  Scalar wPlusI = w + I;
1017  Scalar wPlusI2MC = wPlusI + 2.*ELECTRON_MASS_EV;
1018  Scalar tPlus2MC = eImpact + 2.*ELECTRON_MASS_EV;
1019 
1020  Scalar alpha;
1021  alpha = ELECTRON_MASS_EV / ( eImpact + ELECTRON_MASS_EV );
1022  alpha *= 0.6 * alpha;
1023 
1024  Scalar g2_T_W = sqrt( wPlusI * tPlus2MC / wPlusI2MC / eImpact );
1025  Scalar g3_T_W = alpha * sqrt( I * (1.-wPlusI/eImpact) / w );
1026 
1027  // cout << " g2-1 = " << g2_T_W - 1. << endl;
1028  // cout << " g3 = " << g3_T_W << endl;
1029 
1030  Scalar fTheta = frand();
1031 
1032  thetaPrimary = acos( g2_T_W + g3_T_W * tan(
1033  (1.-fTheta) * atan( (1.-g2_T_W) / g3_T_W )
1034  - fTheta * atan( (1.+g2_T_W) / g3_T_W ) ) );
1035 
1036  // cout << " thetaPrimary = " << thetaPrimary << endl;
1037 
1038  // Second, we generate a Monte Carlo value for phiPrimary --
1039  phiPrimary = TWOPI * frand();
1040 
1041 
1042  // Third, we generate a Monte Carlo value for thetaSecondary --
1043  w = eSecondary;
1044  wPlusI = w + I;
1045  wPlusI2MC = wPlusI + 2.*ELECTRON_MASS_EV;
1046 
1047  g2_T_W = sqrt( wPlusI * tPlus2MC / wPlusI2MC / eImpact );
1048  g3_T_W = alpha * sqrt( I * (1.-wPlusI/eImpact) / w );
1049 
1050  fTheta = frand();
1051 
1052  thetaSecondary = acos( g2_T_W + g3_T_W * tan(
1053  (1.-fTheta) * atan2( (Scalar)(1.-g2_T_W) , (Scalar)g3_T_W )
1054  - fTheta * atan2( (Scalar)(1.+g2_T_W) , (Scalar)g3_T_W ) ) );
1055 
1056  // Fourth, we generate a Monte Carlo value for phiSecondary --
1057  // In the low-energy limit, phiPrimary and phiSecondary are
1058  // uncorrelated. In the high-energy limit, they are 100%
1059  // correlated. Our approach is to implement a sort of sliding
1060  // scale, in which the correlation increases as ePrimary and
1061  // eSecondary both increase.
1062 
1063  Scalar eCritical = 10. * I;
1064  Scalar a_T_W = 0.;
1065  if ( eSecondary>eCritical && ePrimary>eCritical ) {
1066  a_T_W = sqrt((eSecondary-eCritical)*(ePrimary-eCritical))/eCritical;
1067  // cout << " a_T_W = " << a_T_W << endl;
1068  }
1069  Scalar delta = TWOPI/(1.+a_T_W);
1070  // cout << " delta/(2*pi) = " << delta/TWOPI << endl;
1071 
1072  phiSecondary = phiPrimary + M_PI + (frand()-.5)*delta;
1073 
1074  // cout << " phiSecondary-(phiPrimary+PI) = "
1075  // << phiSecondary-phiPrimary-M_PI << endl;
1076

1078 }

1663 /* Method for calculating the e-impact ionization cross section
1664  * for lithium.
1665  *
1666  * The nonrelativistic model uses the fitting function from Eq. 4
1667  * of the 1981 J.Q.S.R.T. paper by S.M. Younger.
1668  *
1669  * The relativistic model uses a parametric function adapted from
1670  * Younger, but modified to capture the relativistic rise. The
1671  * parameters were determined by least squares fit to a) Younger's
1672  * low-energy values and b) the computational results of the LLNL
1673  * report on the EEDL (electron evauated data library), by Perkins,
1674  * Cullen and Seltzer (1991).
1675  *
1676  * Units are eV for energy and m^2 for cross sections.
1677  * NOTE: cross section is NOT in cm^2 and NOT in barns!
1678  *
1679  * First added March 2, 2000, by David Bruhwiler
1680  */
1681 Scalar LiMCC::sigmaIz(Scalar energy) {
1682 
1683  // This is the value that gets returned.
1684  Scalar crossSection;
1685 
1686  // normalized impact energy
1687  // -- assumed that both energy and threshold are in eV
1688  Scalar U = energy / threshold;
1689  if ( U <= 1. ) return 0.;
1690 
1691  // This is needed for both relativistic and NR models
1692  Scalar oneMinusUinv = 1. - 1. / U;
1693 
1694  // nonrelativistic model from Younger's paper
1695  if (staticRelMCC == 0) {
1696  Scalar logU = log(U);
1697  crossSection = youngerParamA * oneMinusUinv
1698  - youngerParamB * oneMinusUinv * oneMinusUinv
1699  + youngerParamC * logU
1700  - youngerParamD * logU / U;
1701  crossSection /= (energy * threshold);
1702  }
1703 
1704  // relativistic model -- fit to Younger and also EEDL calc.'s
1705  else {
1706  Scalar gMin1 = energy / ELECTRON_MASS_EV;
1707  Scalar gamma = gMin1 + 1.;
1708  Scalar beta = sqrt((gamma+1.)*gMin1)/gamma;
1709 
1710  Scalar oneMinusUinvSq = 1.-1./(U*U);
1711 
1712  // "nonrelativistic" energy, scaled to threshold
1713  Scalar Unr = 0.5 * ELECTRON_MASS_EV * beta * beta / threshold;
1714  Scalar logUnr = log(Unr);
1715 
1716  // Now we apply the empirical function to actually calculate
1717  // the cross section:
1718  crossSection = fitParamA1*oneMinusUinv/U
1719  + fitParamA2*logUnr/(U*U)
1720  + fitParamA3*logUnr/Unr
1721  + fitParamA3*oneMinusUinvSq
1722  * ( log( fitParamA4*pow(gamma,fitParamN) ) / Unr
1723  - 2.*threshold/ELECTRON_MASS_EV );
1724 
1725  // cout << "energy sigmaIz = " << energy << "; " << crossSection << endl;
1726  }
1727 
1728  // Return the calculated cross section (units = m^2)
1729  return crossSection;
