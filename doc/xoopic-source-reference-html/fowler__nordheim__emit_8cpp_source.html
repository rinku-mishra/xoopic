204 // computeLocationArray()
205 //
206 // Precomputes a array of equally spaced points mapping the
207 // cumulative distribution to position along the emitter.
208 //
209 // Note that geomFactor includes the radial nature, when the
210 // geometry is cylindrical.
211 //
212 // The integration uses the Trapezoidal Rule.

214 void FowlerNordheimEmitter::computeLocationArray()
215 {
216  Vector2 component(0,0);
217  if (alongx2()) component.set_e2(1.);
218  else component.set_e1(1.);
219 
220  Scalar dx = ( (p2 - p1)*component ) / nIntervals;
221  xArray[0] = p1*component;
222  FArray[0] = 0;
223 
224  // set up for initial pass thru loop
225  Scalar geometryFactor = 1.;
226  if ( alongx2() ) geometryFactor = geomFactor(xArray[0]);
227 
228  Scalar f0, f1;
229  f1 = geometryFactor * getCurrentDensity( xArray[0] );
230 
231  /** Debugging Code */
232  // cerr << "\n\nEntering FowlerNordheimEmitter::computeLocationArray() --" << endl;
233  // cerr << " geometryFactor = " << geometryFactor << endl;
234  // cerr << " xArray[0] = " << xArray[0] << endl;
235  // cerr << " FArray[0] = " << FArray[0] << endl;
236  /** End of Debugging Code */
237 
238  int i;
239  for (i=1; i<=nIntervals; i++) {
240  f0 = f1; // i-1 value
241  xArray[i] = xArray[0] + i * dx;
242 
243  geometryFactor = 1.;
244  if ( alongx2() ) geometryFactor = geomFactor(xArray[i]);
245 
246  f1 = geometryFactor*getCurrentDensity( xArray[i] );
247 
248  FArray[i] = FArray[i-1] + 0.5 * dx * (f0 + f1); // trapezoidal integration
249 
250  /** Debugging Code */

255  /** End of Debugging Code */

257  }

305 Scalar FowlerNordheimEmitter::getCurrentDensity(Scalar _x)
306 {
307  // location at which we wish to know the current density
308  Vector2 positionMKS = p1;
309  if (alongx2()) {
310  positionMKS.set_e2(_x);
311  }
312  else {
313  positionMKS.set_e1(_x);
314  }
315 
316  // normal component of the E field (in V/m) at point of interest
317  // note that fields->E() requires position in grid units!
318  Scalar Efield;
319  Vector3 E = fields->E(fields->getGridCoords(positionMKS));
320  if (alongx2()) Efield = E.e1();
321  else Efield = E.e2();
322 
323  // check the direction of the field to avoid emitting when reversed
324  if (Efield*Boundary::normal >= 0) return 0;
325 
326  Scalar beta_times_E = fabs(beta_FN * Efield);
327 
328  /** Function "y(E)" of the Fowler-Nordheim model.
329  * By default, C_y_FN = 3.79e-05.
330  */
331  y_FN = C_y_FN * sqrt(beta_times_E) / Phi_w_FN;
332 
333  /** Function "v(y)" of the Fowler-Nordheim model
334  * By default, C_v_FN = 0 and v_y_FN = 1.
335  */
336  v_y_FN = 1. - C_v_FN * y_FN*y_FN;
337 
338  // current density J_FN (in A/m^2) at point of interest
339  Scalar factor_FN;
340  Scalar exp_factor_tmp;
341  Scalar exp_factor_FN;
342  Scalar currentDensity_FN;
343  factor_FN = A_FN * beta_times_E * beta_times_E / Phi_w_FN;
344 
345  exp_factor_tmp = B_FN * v_y_FN * pow(Phi_w_FN, Scalar(1.5));
346  if ( exp_factor_tmp < beta_times_E * Scalar_MAX ) {
347  exp_factor_FN = exp_factor_tmp / beta_times_E;
348  }
349  else {
350  exp_factor_FN = Scalar_MAX;
351  }
352  if ( exp_factor_FN < log(1./Scalar_MIN) ) {
353  currentDensity_FN = factor_FN * exp(-exp_factor_FN);
354  }
355  else {
356  currentDensity_FN = 0.;
357  }
358 
359  /** Debugging Code */
360  // cerr << "\n\nEntering FowlerNordheimEmitter::getCurrentDensity(x) --" << endl;
361  // cerr << " _x = " << _x << endl;
362  // cerr << " _y = " << positionMKS.e2() << endl;
363  // cerr << " Efield = " << Efield << endl;
