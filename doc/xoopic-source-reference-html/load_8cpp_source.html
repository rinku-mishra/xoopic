272 void Load::init_fajans_load()
273 throw(Oops){
274 #define MAX_ITER 50
275  int j, k, i;
276  /* new code for smart loading - djc */
277 
278  rmax_fn_z = new Scalar[J+1];
279  Scalar interpolation;
280  Scalar constant;
281  Scalar dr, f, df;
282  int j1 =0;
283  /* I assume J=jmax and j= 0 (cells) are boundaries where the plasma density
284  may be assumed nil, and the potential is given very nearly by
285  the vaccuum solution - djc */
286  /* I assume getphi()[z][r] */
287  constant = density*(q)*iEPS0/4.0; /* positron charge assumed */
288 
289  // run a zero souce though dadi to get the vaccuum solution
290  region->ComputePhi();
291 
292  Scalar equi_pot = fields->getphi()[j1][0];
293 
294  /*iterate and generate rmax_fn_z here using small r approx. for first guess */
295  /* numerical recipes Newton-Raphson method*/
296 
297  rmax_fn_z[j1] = 0.0;
298  for (k= (j1 + 1); k<J; k++) {
299  if ( k == (j1 + 1)) {
300  rmax_fn_z[k] = sqrt((equi_pot - fields->getphi()[k][0])/2.0/constant);
301  }
302  else {
303  rmax_fn_z[k] = rmax_fn_z[k-1];
304  }
305  for(i=1; i<= MAX_ITER; i++)
306  {
307  f = constant*rmax_fn_z[k]*rmax_fn_z[k]*(1 + 2.0*log(Max2MKS/rmax_fn_z[k])) - (equi_pot - fields->getphi()[k][0]);
308  df = 4.0*constant*rmax_fn_z[k]*log(Max2MKS/rmax_fn_z[k]);
309  dr = f/df;
310  rmax_fn_z[k] -= dr;
311 
312  if ((Max2MKS - rmax_fn_z[k])*(rmax_fn_z[k]) < 0.0) {
313  stringstream ss (stringstream::in | stringstream::out);
314  ss << "Load::init_fajans_load: Error: \n"<<
315  "jumped out of bounds in load " << endl;
316 
317  std::string msg;
318  ss >> msg;
319  Oops oops(msg);
320  throw oops; // exit() //Load::Load
321 
322  }
323  if (fabs(dr/rmax_fn_z[k]) <= 0.01) break; /* 1 percent tollerance imposed */
324  }
325  if (fabs(dr/rmax_fn_z[k]) > 0.01)
326  {
327  cout << "max number of iterations reached in load " << endl;
328  }
329  }
330  rmax_fn_z[J] = 0.0;
331  Scalar temp_density = 0;
332  for (j=0; j<=J; j++)
333  {
334  for (k=0; k<=K; k++)
335  {
336  /* interpolation */
337  interpolation = rmax_fn_z[j];
338  if ((grid->getMKS(j,k).e2()) > interpolation)
339  {
340  temp_density=0;
341  }
342  else
343  {
344  temp_density = 1;
345  }
346  gridded_density[j][k] = temp_density;
347  fields->setloaddensity(species->getID(),j,k,temp_density);
348  }
349  }
