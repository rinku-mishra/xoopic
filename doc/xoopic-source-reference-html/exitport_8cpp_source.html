35  iCeq = 1/C;

164 void ExitPort::applyFields(Scalar t,Scalar dt)
165 {
166 
167  if(!t) return; // don't need to do this at t=0, crashes in fact.
168  if (init) initialize(t,dt);
169  time = t - dt/2;
170  // cout << "time " << time << endl;
171  Scalar H3, dE2dl, E2dl, HonBoundary, Stuff, H2, dE3dl;
172  Scalar EonBoundary, E3dl, H1, dE1dl, E1dl;
173  Scalar E2origdl, E3origdl, E1origdl;
174  int Kdex, Jdex;
175 
176  if (fields->getSub_Cycle_Iter() ==1) EnergyOut = 0.0;
177  else if (fields->getSub_Cycle_Iter() ==0) EnergyOut = 0.0;
178 
179  // EnergyOut = 0.0;
180 
181  if (alongx2()) // vertical
182  {
183  // wave going to the right normal = -1, shift = -1
184  // wave going to the left normal = 1, shift = 0
185  for (int k = k1; k < k2; k++)
186  {
187  Kdex = k-k1;
188  // TM mode
189  E2origdl = IntEdl[j2][k].e2();
190  H3 = (iL[j2+shift][k].e3())*(IntBdS[j2+shift][k].e3());
191  Stuff = 2*H3 + normal*I[j2][k].e2();
192  dE2dl = normal*(Rprime*Stuff - 2*Lprime*oldHonBoundary[Kdex].e1())
193  - 2*SourceTM(j2,k,Kdex);
194  E2dl = (minusTM[Kdex]*E2origdl - dE2dl)*iplusTM[Kdex];
195  dE2dl = E2dl - E2origdl;
196  EonBoundary = .5*(E2dl + E2origdl)/grid->dl2(j2,k);
197  HonBoundary = .5*(Stuff + normal*idtCTM[Kdex]*dE2dl);
198  fields->setIntEdl(j2, k, 2, E2dl);
199  IntTM[Kdex] += iCdt2*(HonBoundary + oldHonBoundary[Kdex].e1());
200  oldHonBoundary[Kdex].set_e1(HonBoundary);
201  Scalar g = grid->dl3(j2+shift,k);
202  if (g!=0)
203  EnergyOut += normal*EonBoundary*HonBoundary*grid->dS(j2+shift,k).e1()/g;
204 
205  // TE mode
206  E3origdl = IntEdl[j2][k].e3();
207  H2 = (iL[j2+shift][k].e2())*(IntBdS[j2+shift][k].e2());
208  if (k==0)
209  {
210  Stuff = normal*I[j2][k].e3() + 2*H2;
211  }
212  else
213  {
214  Stuff = normal*(-iL[j2][k].e1()*IntBdS[j2][k].e1() +
215  iL[j2][k-1].e1()*IntBdS[j2][k-1].e1() -
216  I[j2][k].e3()) + 2*H2;
217  }
218  dE3dl = normal*(Rprime*Stuff + 2*Lprime*oldHonBoundary[Kdex].e2())
219  - 2*SourceTE(j2,k,Kdex);
220  E3dl = (minusTE[Kdex]*E3origdl + dE3dl)*iplusTE[Kdex];
221  dE3dl = (E3dl - E3origdl);
222  HonBoundary = .5*(Stuff - normal*idtCTE[Kdex]*dE3dl);
223  g = grid->dl3(j2,k);
224  if (g!=0)
225  EonBoundary = .5*(E3dl + E3origdl)/g;
226  else
227  EonBoundary = 0;
228  fields->setIntEdl(j2, k, 3, E3dl);
229  IntTE[Kdex] += iCdt2*(HonBoundary + oldHonBoundary[Kdex].e2());
230  oldHonBoundary[Kdex].set_e2(HonBoundary);
231  g = grid->dl2(j2+shift,k);
232  if (g!=0)
233  EnergyOut -= normal*EonBoundary*HonBoundary*grid->dS(j2+shift,k).e1()/g;
234  }
235  }
236  else // horizontal
237  {
238  // wave going in (smaller y) normal = +1, shift = 0
239  // wave going out (bigger y) normal = -1, shift = -1
240  for (int j=MIN(j1,j2); j<MAX(j1,j2); j++)
241  {
242  Jdex = j-j1;
243  // TM mode
244  E1origdl = IntEdl[j][k2].e1();
245  H3 = iL[j][k2+shift].e3()*IntBdS[j][k2+shift].e3();
246  Stuff = 2*H3-normal*I[j][k2].e1();
247  dE1dl = normal*(Rprime*Stuff - 2*Lprime*oldHonBoundary[Jdex].e1()) - 2*SourceTM(j,k2,Jdex);
248  E1dl = (minusTM[Jdex]*E1origdl + dE1dl)*iplusTM[Jdex];
249  dE1dl = E1dl - E1origdl;
250  EonBoundary = .5*(E1origdl+E1dl)/grid->dl1(j,k2);
251  HonBoundary = .5*(Stuff - normal*idtCTM[Jdex]*dE1dl);
252  fields->setIntEdl(j, k2, 1, E1dl);
253  oldHonBoundary[Jdex].set_e1(HonBoundary);
254  EnergyOut -= normal*EonBoundary*HonBoundary*grid->dS(j,k2+shift).e2()/
255  (grid->dl(j,k2+shift).e3());
256 
257  // TE mode
258  if (j==0) fields->setIntEdl(0, k2, 3, 0);
259  else
260  {
261  E3origdl = IntEdl[j][k2].e3();
262  H1 = (iL[j][k2+shift].e1())*(IntBdS[j][k2+shift].e1());
263  Stuff = 2*H1
264  -normal*(iL[j][k2].e2()*IntBdS[j][k2].e2() -
265  iL[j-1][k2].e2()*IntBdS[j-1][k2].e2() +
266  I[j][k2].e3());
267  dE3dl = -normal*(Rprime*Stuff - 2*Lprime*oldHonBoundary[Jdex].e2()) - 2*SourceTE(j,k2,Jdex);
268  E3dl = (minusTE[Jdex]*E3origdl + dE3dl)*iplusTE[Jdex];
269  dE3dl = E3dl - E3origdl;
270  EonBoundary = .5*(E3origdl + E3dl)/grid->dl3(j,k2);
271  HonBoundary = .5*(Stuff + normal*idtCTE[Jdex]*dE3dl);
272  fields->setIntEdl(j, k2, 3, E3dl);
273  oldHonBoundary[Jdex].set_e2(HonBoundary);
274  EnergyOut += normal*EonBoundary*HonBoundary*grid->dS(j,k2+shift).e2()/
275  (grid->dl(j,k2+shift).e1());
276  }
277  }
278  }
279  if (fields->getFieldSub() == fields->getSub_Cycle_Iter()) EnergyOut /= (fields->getFieldSub());

282 void ExitPort::initialize(Scalar t,Scalar dt)
283 {
284  grid = fields->get_grid();
285  iL = fields->getiL();
286  IntBdS = fields->getIntBdS();
287  IntEdl = fields->getIntEdl();
288  I = fields->getI();
289  Lprime = L/dt;
290  normal = get_normal();
291  if (normal == 1) shift = 0;
292  else shift = -1;
293  if (C)
294  {
295  Rprime = R+Lprime+dt/(4*C);
296  iCdt2 = dt/(2*C);
297  Lprime -= iCdt2;
298  }
299  else
300  {
301  Rprime = R+Lprime;
302  iCdt2 = 0;
303  }
304 
305  Scalar RidtC, dl3local;
306  int Jdex, Kdex;
307 
308  if (alongx2()) // vertical
309  {
310  // wave going to the right normal = -1
311  // wave going to the left normal = 1
312  for (int k = k1; k < k2; k++)
313  {
314  Kdex = k-k1;
315  // TM mode
316  if (grid->query_geometry())
317  dl3local = 1;
318  else
319  dl3local = 2*M_PI*(grid->getMKS(Vector2(j1+shift, k + .5))).e2();
320  ATM[Kdex] = dl3local/(grid->dl2(j1,k));
321  if(fields->getiC()[j2+normal][k].e2())
322  {
323  idtCTM[Kdex] = 1/(dt*fields->getiC()[j2+normal][k].e2()); // iC j2 is equal to zero because of the passive B.C.
324  RidtC = Rprime*idtCTM[Kdex];
325  iplusTM[Kdex] = 1/(ATM[Kdex]+RidtC);
326  minusTM[Kdex] = RidtC-ATM[Kdex];
327  }
328  else
329  idtCTM[Kdex] = iplusTM[Kdex] = minusTM[Kdex] = FALSE;
330 
331  // TE mode
332  if (grid->query_geometry())
333  dl3local = 1;
334  else
335  dl3local = 2*M_PI*(grid->getMKS(Vector2(j1, k + .5))).e2();
336  ATE[Kdex] = (grid->dl2(j1,k))/dl3local;
337  if(fields->getiC()[j2+normal][k].e3())
338  {
339  idtCTE[Kdex] = 1/(dt*fields->getiC()[j2+normal][k].e3());//iC on the edge is set to zero
340  RidtC = Rprime*idtCTE[Kdex];
341  iplusTE[Kdex] = 1/(ATE[Kdex]+RidtC);
342  minusTE[Kdex] = RidtC-ATE[Kdex];
343  }
344  else
345  idtCTE[Kdex] = iplusTE[Kdex] = minusTE[Kdex] = FALSE;
346  }
347  }
348  else // horizontal
349  {
350  // wave going out (greater r) normal = -1
351  // wave going in normal = 1
352  for (int j=j1; j<j2; j++)
353  {
354  Jdex= j-j1;
355  // TM mode
356  if (grid->query_geometry())
357  dl3local = 1;
358  else
359  dl3local = 2*M_PI*(grid->getMKS(Vector2(j, k2+normal/2))).e2();
360  ATM[Jdex] = dl3local/(grid->dl1(j,k2));
361  if (fields->getiC()[j][k2+normal].e1())
362  {
363  idtCTM[Jdex] = 1/(dt*(fields->getiC()[j][k2+normal].e1())); // iC k2 is equal to zero because of the passive B.C.
364  RidtC = Rprime*idtCTM[Jdex];
365  iplusTM[Jdex] = 1/(RidtC+ATM[Jdex]);
366  minusTM[Jdex] = RidtC-ATM[Jdex];
367  }
368  else
369  idtCTM[Jdex] = iplusTM[Jdex] = minusTM[Jdex] = FALSE;
370 
371  // TE mode
372  if (grid->query_geometry())
373  dl3local = 1;
374  else
375  dl3local = 2*M_PI*(grid->getMKS(Vector2(j, k2 + .5))).e2();
376  ATE[Jdex] = (grid->dl1(j,k2))/dl3local;
377  if (fields->getiC()[j][k2+normal].e3())
378  {
379  idtCTE[Jdex] = 1/(dt*(fields->getiC()[j][k2+normal].e3()));
380  RidtC = Rprime*idtCTE[Jdex];
381  iplusTE[Jdex] = 1/(RidtC+ATE[Jdex]);
382  minusTE[Jdex] = RidtC-ATE[Jdex];
383  }
384  else
385  idtCTE[Jdex] = iplusTE[Jdex] = minusTE[Jdex] = FALSE;
386  }
387  }
388  init = FALSE;
389 }
