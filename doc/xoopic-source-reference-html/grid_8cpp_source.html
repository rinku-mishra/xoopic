186 // Applies a binomial filter (4, 2, 1) in 2d to filter the short wavelengths
187 // in the arg array. n specifies the number of times to apply the filter.
188 // In order to do this in place (i.e. avoid a scratch array) the filter is performed
189 // in four passes (B&L, Appendix C). The array is assumed to be on the grid,
190 // with dimensions JxK. The smoothing presently does not move charge off
191 // boundary surfaces.
192 
193 // need to fix for periodic bc
194 
195 void Grid::binomialFilter(Scalar **array, int n)
196 {
197  int i, j, k;
198  for (i=0; i<n; i++){
199  for (j=0; j<=J; j++) // +x1 pass
200  for (k=0; k<=K; k++)
201  if (j==J || (CellMask[j][k] && CellMask[j][MAX(0,k-1)]))
202  array[j][k] += array[j][k]; // so we can divide by 16 later
203  else array[j][k] += array[j+1][k];
204  for (j=J; j>=0; j--) // -x1 pass
205  for (k=0; k<=K; k++)
206  if (j==0 || (CellMask[j-1][k] && CellMask[j-1][MAX(0,k-1)]))
207  array[j][k] += array[j][k]; // so we can divide by 16 later
208  else array[j][k] += array[j-1][k];
209  for (j=0; j<=J; j++) // +x2 pass
210  for (k=0; k<=K; k++)
211  if (k==K || (CellMask[j][k] && CellMask[MAX(0,j-1)][k]))
212  array[j][k] += array[j][k]; // so we can divide by 16 later
213  else array[j][k] += array[j][k+1];
214  for (j=0; j<=J; j++) // -x2 pass
215  for (k=K; k>=0; k--){
216  if (k==0 || (CellMask[j][k-1] && CellMask[MAX(0,j-1)][k-1]))
217  array[j][k] += array[j][k]; // so we can divide by 16 later
218  else array[j][k] += array[j][k-1];
219  array[j][k] *= 6.25e-2;
220  }
221  if (PeriodicFlagX1){ // average along periodic boundaries
222  for (k=0; k<=K; k++) {
223  array[0][k] += array[J][k];
224  array[0][k] *= 0.5;
225  array[J][k] = array[0][k]; // synchronize
226  }
227  }
228  if (PeriodicFlagX2){
229  for (j=0; j<=J; j++) {
230  array[j][0] += array[j][K];
231  array[j][0] *= 0.5;
232  array[j][K] = array[j][0];
233  }
234  }
235  }
