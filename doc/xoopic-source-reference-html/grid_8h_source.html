258 //--------------------------------------------------------------------
259 // Given the initial position (x0), velocity v, normalized
260 // momentum u = gamma*v, and timestep dt,
261 // compute the differential position and return it. The present
262 // routine is for z-r-phi coords only. All units are MKS. A side
263 // effect of this routine is the rotation of v to the new local
264 // coordinate system of the particle.
265 
266 inline Vector3 Grid::differentialMove(const Vector2& x0,const Vector3& v, Vector3& u,
267  Scalar dt)
268 {
269  switch(geometryFlag)
270  {
271  case ZXGEOM:
272  return Vector3(v.e1()*dt, v.e2()*dt, v.e3()*dt);
273 
274  case ZRGEOM:
275  default:
276  {
277  Vector3 xf(x0.e1() + v.e1()*dt, x0.e2() + v.e2()*dt, v.e3()*dt);
278  // get new radius:
279  Scalar r2 = sqrt(xf.e2()*xf.e2() + xf.e3()*xf.e3());
280 
281  // alpha is the angle of rotation, phi2 - phi1, for this timestep
282  Scalar sinAlpha;
283  Scalar cosAlpha;
284  if (r2 > 0)
285  {
286  sinAlpha = xf.e3()/r2;
287  cosAlpha = xf.e2()/r2;
288  }
289  else
290  {
291  sinAlpha = 0;
292  cosAlpha = 1;
293  }
294 
295  // rotate u into new coordinate system:
296  Scalar u2r = cosAlpha*u.e2() + sinAlpha*u.e3();
297  Scalar u2phi = -sinAlpha*u.e2() + cosAlpha*u.e3();
298  u.set_e2(u2r);
299  u.set_e3(u2phi);
300 
301  // compute and return dx:
302  return Vector3(xf.e1() - x0.e1(), r2 - x0.e2(), xf.e3());
303  }
304  }
305 }
