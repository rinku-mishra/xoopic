247 void DielectricRegionParams::checkRules()
248 {
249  // Rules from BoundaryParams
250  ostring result = OnGridConstraint();
251  if (strlen(result.c_str()) > 0) ruleMessages.add(new ostring(result));
252  // Loop thru rules
253  oopicListIter<RuleBase> nR(RuleList);
254  oopicListIter<BaseParameter> nP(parameterList);
255  for (nR.restart(); !nR.Done(); nR++)
256  // Loop thru parameters and set parameter values required by rule
257  {
258  for (nP.restart(); !nP.Done(); nP++)
259  nR.current()->setRuleVariable(nP.current()->getName(),

298 // I basically have to clip the "lines" defined by the coordinates
299 // given.

305  S->A = Vector2((G->MaxMPIx1-G->MinMPIx1)*S->A.e1()/(Scalar)G->MaxMPIJ,
