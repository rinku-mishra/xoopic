97 void PDistCol::record_particle(Particle &p, int component)
98 {
99  int spec = p.get_speciesID();
100  //collect the charge of this particle
101  Qcol+=p.get_q();
102  Qcol_sp[spec] += p.get_q(); // also save by species
103 
104  if(!nxbins) return;
105  Scalar xp = (component)? p.get_x().e2(): p.get_x().e1();
106  xp = (xp-x_min)*xscale;
107  if (xp > nxbins+1 || xp < 0) return;
108 
109  /* actually do the accumulation! */
110  /* this assumes that the particle has the position on the boundary given by xp*/
111  int j = (int) xp; // the floor index
112  Scalar w = xp - j; // the fraction within the bin
113  fxdata[spec][j] += p.get_np2c()*(1-w);
114  fxdata[spec][j+1] += p.get_np2c()*w;
115 
116  if (nenergybins){
117  Scalar e = (p.energy_eV_MKS() - energy_min)*escale;
118  if (e > 0 && e < nenergybins) {
119  int k = (int) e;
120  Scalar we = e - k; // fraction in bin
121  if (w){
122  fedata[spec][j][k] += p.get_np2c()*(1-w)*(1-we);
123  fedata[spec][j][k+1] += p.get_np2c()*(1-w)*we;
124  fedata[spec][j+1][k] += p.get_np2c()*w*(1-we);
125  fedata[spec][j+1][k+1] += p.get_np2c()*w*we;
126  }
127  else{
128  fedata[spec][j][k] += p.get_np2c()*(1-we);
129  fedata[spec][j][k+1] += p.get_np2c()*we;
130  }
131  }
132  }
133 
134  if (nthetabins){
135  Scalar t;
136  Vector3 u = p.get_u();
137  if (component) t = atan(sqrt(u.e2()*u.e2() + u.e3()*u.e3())/u.e1());
138  else t = atan(sqrt(u.e1()*u.e1() + u.e3()*u.e3())/u.e2());
139  t = (t - theta_min)*tscale;
140  if (t > 0 && t < nthetabins){
141  int k = (int) t;
142  Scalar wt = t - k;
143  if (w){
144  ftdata[spec][j][k] += p.get_np2c()*(1-w)*(1-wt);
145  ftdata[spec][j][k+1] += p.get_np2c()*(1-w)*wt;
146  ftdata[spec][j+1][k] += p.get_np2c()*w*(1-wt);
147  ftdata[spec][j+1][k+1] += p.get_np2c()*w*wt;
148  }
149  else{
150  ftdata[spec][j][k] += p.get_np2c()*(1-wt);
151  ftdata[spec][j][k+1] += p.get_np2c()*wt;
152  }
153  }
154  }
155 }
