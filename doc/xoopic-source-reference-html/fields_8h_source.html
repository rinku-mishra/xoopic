353 // Accumulate current based on x(t) and x(t+dt*f). Here frac_v3dt
354 // is the fraction of the timestep for this piece of the move times
355 // the azimuthal displacement v3*dt. Cell volume is more accurately
356 // chosen separately for each current element.
357 // q_dt == q/dt
358 
359 inline void Fields::accumulate(Vector2& xOld, Vector2&x, Scalar frac_v3dt,
360  Scalar q_dt)
361 {
362  if (get_freezeFields()) return; // emit always does EM push
363  int j = (int) MIN(xOld.e1(),x.e1());
364  int k = (int) MIN(xOld.e2(),x.e2());
365  if (j == grid->getJ()) j--;
366  if (k == grid->getK()) k--;
367  Vector2 dw = x - xOld;
368  Vector2 avg_w = 0.5*(xOld + x) - Vector2(j, k);
369 
370  frac_v3dt /= grid->cellVolume(j, k);
371  if(CurrentWeightingFlag==0)
372  {
373  if(accumulator) {
374  accumulator[j][k] += q_dt*Vector3(dw.e1()*(1 - avg_w.e2()),
375  dw.e2()*(1 - avg_w.e1()),
376  grid->dS3Prime(j, k)*(1 - avg_w.e1())*
377  (1 - avg_w.e2())*frac_v3dt);
378 
379  accumulator[j][k+1] += q_dt*Vector3(dw.e1()*avg_w.e2(),
380  0,
381  grid->dS3Prime(j, k+1)*
382  (1 - avg_w.e1())*avg_w.e2()*frac_v3dt);
383  accumulator[j+1][k] += q_dt*Vector3(0,
384  dw.e2()*avg_w.e1(),
385  grid->dS3Prime(j+1, k)*
386  avg_w.e1()*(1 - avg_w.e2())*frac_v3dt);
387 
388  accumulator[j+1][k+1] += q_dt*Vector3(0,
389  0,
390  grid->dS3Prime(j+1, k+1)*avg_w.e1()*avg_w.e2()
391  *frac_v3dt);
392  }
393  }
394  else /* the bilinear weighting of current */
395  {
396  if(accumulator) {
397  accumulator[j][k] += q_dt*Vector3(dw.e1()*(1 - avg_w.e2())*(1-avg_w.e1()),
398  dw.e2()*(1 - avg_w.e1())*(1-avg_w.e2()),
399  grid->dS3Prime(j, k)*(1 - avg_w.e1())*
400  (1 - avg_w.e2())*frac_v3dt);
401 
402  accumulator[j][k+1] += q_dt*Vector3(dw.e1()*avg_w.e2()*(1-avg_w.e1()),
403  dw.e2()*avg_w.e2()*(1-avg_w.e1()),
404  grid->dS3Prime(j, k+1)*
405  (1 - avg_w.e1())*avg_w.e2()*frac_v3dt);
406 
407  accumulator[j+1][k] += q_dt*Vector3(dw.e1()*avg_w.e1()*(1-avg_w.e2()),
408  dw.e2()*avg_w.e1()*(1-avg_w.e2()),
409  grid->dS3Prime(j+1, k)*
410  avg_w.e1()*(1 - avg_w.e2())*frac_v3dt);
411  accumulator[j+1][k+1] += q_dt*Vector3(dw.e1()*avg_w.e1()*avg_w.e2(),
412  dw.e2()*avg_w.e1()*avg_w.e2(),
413  grid->dS3Prime(j+1, k+1)*avg_w.e1()*avg_w.e2()
414  *frac_v3dt);
415  }
416  }
