141 Load* LoadParams::CreateCounterPart(SpatialRegion * SP)
142  throw(Oops) {
143  region = SP;
144  Load* loader;
145  Species* species = spatialRegionGroup->get_speciesPtr(
146  speciesName.getValue());
147  Maxwellian* maxwellian;
148 
149  UNIT type=MKS;
150  if (units.getValue()==(ostring)"MKS") type = MKS;
151  else if (units.getValue()==(ostring)"EV") type = EV;
152  else {
153  std::string msg = "LoadParams::CreateCounterPart: units '";
154  msg += units.getValue().c_str();
155  msg += "' unknown.";
156  Oops oops(msg);
157  throw oops;
158  }
159  maxwellian = new Maxwellian(species);
160  if (temperature.getValue()==0) maxwellian->setThermal(
161  Vector3(v1thermal.getValue(),v2thermal.getValue(),
162  v3thermal.getValue()),type);
163  else maxwellian->setThermal(temperature.getValue(),type);
164 
165  if ((Ucutoff.getValue()==0)&&(Lcutoff.getValue()==0))
166  maxwellian->setCutoff(Vector3(v1Lcutoff.getValue(),v2Lcutoff.getValue(),
167  v3Lcutoff.getValue()),
168  Vector3(v1Ucutoff.getValue(),v2Ucutoff.getValue(),
169  v3Ucutoff.getValue()),type);
170  else
171  maxwellian->setCutoff(Lcutoff.getValue(),Ucutoff.getValue(),type);
172 
173  maxwellian->setDrift(Vector3(v1drift.getValue(), v2drift.getValue(),
174  v3drift.getValue()),type);
175  try{
176  loader = new Load(region,density.getValue(), maxwellian,
177  x1MinMKS.getValue(), x1MaxMKS.getValue(),
178  x2MinMKS.getValue(), x2MaxMKS.getValue(),
179  LoadMethodFlag.getValue(),
180  np2c.getValue(),analyticF.getValue(),
181  densityfile.getValue(),Name.getValue());
182  }
183  catch(Oops& oops){
184  oops.prepend("LoadParams::CreateCounterPart: Error: \n"); // not called
185  throw oops;
186  }
187 
188 
189  if (!WasDumpFileGiven) {
190  try{
191  loader->load_it();
192  }
193  catch(Oops& oops){
194  oops.prepend("LoadParams::CreateCounterPart: Error: \n"); //OK
195  throw oops;
196  }
197  }
198 
199  region->theLoads->add(loader);
200  //delete maxwellian;
201  return loader;
