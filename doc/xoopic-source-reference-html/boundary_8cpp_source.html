430 int Boundary::onMe(Scalar tA1,Scalar tA2,Scalar tB1,Scalar tB2) {
431  Grid *grid = fields->get_grid();
432  Scalar A1 = grid->getX()[j1][k1].e1();
433  Scalar A2 = grid->getX()[j1][k1].e2();
434  Scalar B1 = grid->getX()[j2][k2].e1();
435  Scalar B2 = grid->getX()[j2][k2].e2();
436 
437  // First check to see if it can overlap the current boundary
438  if(MIN(tA1,tB1) > MAX(A1,B1)) // there is no X1 overlap
439  return 0;
440  if(MAX(tA1,tB1) < MIN(A1,B1)) // there is no X1 overlap
441  return 0;
442  if(MIN(tA2,tB2) > MAX(A2,B2)) // there is no X2 overlap
443  return 0;
444  if(MAX(tA2,tB2) < MIN(A2,B2)) // there is no X2 overlap
445  return 0;
446  // pick a minimum delta: if the boundary read in has distances less
447  // than delta, assume it's the same boundary.
448  Scalar delta = 0.001*MIN(-grid->getX()[0][0].e2() + grid->getX()[0][1].e2(),
449  -grid->getX()[0][0].e1() + grid->getX()[1][0].e1());
450 
451  Scalar Dist1 = LineDist(A1,A2,B1,B2,tA1,tA2);
452  Scalar Dist2 = LineDist(A1,A2,B1,B2,tB1,tB2);
453 
454  // if both distances are less than Delta, it's in this boundary.
455  if(Dist1 < delta && Dist2 < delta) { return 1;}
456  else return 0;
457 }
