172 void WaveDirDiag::MaintainDiag(Scalar t, int n)
173 {
174  int j,k;
175  Grid *grid = region->get_grid();
176  Vector3 **E = region->getIntEdl();
177  Vector3 **B = region->getIntBdS();
178  /* Vector3 **E = region->getENode();
179  Vector3 **B = region->getBNode(); */
180  //
181  // components of the E and B fields for which to
182  // calculate the linear compbinations, alpha has
183  // the value given by the Ecomponent label and
184  // beta the one given by the Bcomponent.
185  //
186  Scalar Ealpha;
187  Scalar Bbeta;
188 
189  double n0 = SPEED_OF_LIGHT;
190 
191  for(j=1;j<xsize-3;j++)
192  for(k=0;k<ysize;k++) {
193  //
194  // we use the fact that the moving window requires a uniform
195  // grid spacing and we make the same assumption here and use
196  // the second order Lagrange interpolation with equal distance
197  // between the points.
198  //
199  Ealpha =
200  aj*E[j][k].e(Ecomponent) +
201  ajp1*E[j+1][k].e(Ecomponent) +
202  ajp2*E[j+2][k].e(Ecomponent);
203 
204  Ealpha /= (grid->dl(j,k)).e(Ecomponent);
205  Bbeta = B[j][k].e(Bcomponent)/(grid->dS(j,k)).e(Bcomponent);
206 
207  Rdata[j][k] = 0.5 *(Ealpha + signRdata * n0 * Bbeta);
208  Ldata[j][k] = 0.5 *(Ealpha + signLdata * n0 * Bbeta);
209  }
210 #ifdef HAVE_FFT
211  if ( psd1dFlag ) {
212  get_psd1d_Diagnostics(Rdata, Rdata_psd1d);
213  get_psd1d_Diagnostics(Ldata, Ldata_psd1d);
214  }
215 #endif // HAVE_FFT
216 }
