349 // Advance particle equations of motion one timestep. This includes

351 // and accumulating the particle current on the grid. Boris mover
352 // with u = gamma*v, v in MKS units; see B&L for variables.

354 void ParticleGroup::advance(Fields& fields, Scalar dt)
355 {
356  Scalar f = 0.5*dt*q_over_m;
357  Scalar q_dt = q/dt;
358  Vector3 uPrime;
359  Vector3 a;
360  Vector3 t;
361  Vector3 s;
362  Vector2* x = ParticleGroup::x;
363  Vector3* u = ParticleGroup::u;
364  Vector3 dxMKS;
365  Boundary* bPtr;
366  Grid& grid = *fields.get_grid();
367  Scalar igamma;
368  Scalar localEnergy = 0;
369 
370  for (int i=0; i<n; i++, x++, u++)
371  {
372  if (qArray) q_dt = qArray[i]/dt; // variable weighted particles
373  a = f*fields.E(*x);
374 //
375  *u += a; // half acceleration
376  igamma = 1/ParticleGroup::gamma(*u);
377  t = (f*igamma)*fields.B(*x);
378  uPrime = *u + u->cross(t);
379  s = 2*t/(1 + t*t);
380  *u += uPrime.cross(s); // rotation
381  *u += a; // half acceleration
382  igamma = 1/ParticleGroup::gamma(*u); // must recompute gamma
383  dxMKS = grid.differentialMove(fields.getMKS(*x), *u*igamma, *u, dt);
384  if ((bPtr = fields.translateAccumulate(*x, dxMKS, q_dt)))
385  {
386  // send this particle to boundary
387  bPtr->collect(*(new Particle(*x, *u, species, get_np2c(i),
388  (BOOL)(qArray!=0))), dxMKS);
389  // Move last particle into this slot and advance it.
390  n--;
391  if (i == n) break; // last particle in array?
392  *x = ParticleGroup::x[n]; // move last particle into open slot
393  *u = ParticleGroup::u[n];
394  if (qArray) qArray[i] = qArray[n];
395  i--; // Set index to do i again.
396  x--; // Setup to redo the array element
397  u--; // it gets inc in for loop
398  }
399  else{
400  if (qArray)
401  localEnergy += qArray[i]*(1/igamma-1);
402  else
403  localEnergy += 1/igamma-1;
404  }
405 #ifdef DEBUG
406  if(strncmp(HOSTTYPE, "alpha", 5) != 0)
407  {
408  if(((*x).e1()==grid.getJ() || (*x).e1()==0 || (*x).e2()==grid.getK()
409  || (*x).e2()==0)&&!bPtr) {
410  printf("Particle escaped.\n");
411  int *crash = 0;
412  *crash = 10;
413  }
414  }
415 #endif //DEBUG
416  }
417  if (qArray)
418  species->addKineticEnergy(SPEED_OF_LIGHT_SQ*localEnergy/q_over_m);
419  else
420  species->addKineticEnergy(SPEED_OF_LIGHT_SQ*localEnergy*get_m());
421 }

427 void ParticleGroup::increaseParticleWeight()
428 {
429  int new_n = (int) (n*0.5 + frand()); // keep odd particles 50% time
430  Scalar ratio = 2; // = ((Scalar) n)/new_n;
431  if (qArray) qArray[0] *= ratio;

433  {
434  x[i]=x[2*i];

436  if(qArray) qArray[i] = qArray[2*i]*ratio;

438  n = new_n; // halve the num of particles
439  np2c0 *= ratio; // double the weights
440  q *= ratio; // double the charge
