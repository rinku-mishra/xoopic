381 /**********************************************************************
382  * a helper function to calculate averages and standard deviations
383  * for an array of "random" variables distributed according to
384  * the array of weight elements _Pq. It uses a two-pass algorithm.
385  * DAD: Thu Oct 19 11:23:14 MDT 2000
386  **********************************************************************/
387 void Diagnostics::Get_Statistical_Moments(int N, Scalar* W, Scalar* Data,
388  Scalar& Ave, Scalar& Var)
389 {
390  // first calculate the average
391  Ave = 0.0;
392  int i;
393  for ( i = 0; i < N; i++ )
394  Ave += W[i] * Data[i];
395  // now calculate the variance on the second pass.
396  Scalar tmp;
397  Scalar sum = 0.0;
398  Scalar sum2 = 0.0;
399  for ( i = 0; i < N; i++ ) {
400  tmp = Data[i] - Ave;
401  sum += W[i] * tmp;
402  sum2 += W[i] * tmp * tmp;
403  }
404 
405  Var = sum2 - (sum * sum);
406  // check if it is positive
407  if ( Var < 0.0 ) {
408  if ( Var > -Scalar_EPSILON ) { // a precision problem set Var to zero
409  Var= 0.0;
410  } else {
411  printf(" ");
412  printf("WARNING: Negative variance found in ");
413  printf("Diagnostics::Get_Statistical_Moments(...)");
414  printf("This should NEVER happen!!!");
415  printf("Setting the value of the variance to ZERO!");
416  }
417  }
418 
419  return;
420 }
