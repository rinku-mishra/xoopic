246  int segcount = 0; /* number of segments. */
247  /* reorder the input such that jl,kl is the leftmost point */
248  if(j1<j2) {
249  jl=j1;kl=k1;
250  jh=j2;kh=k2;
251  }
252  else {
253  jl=j2;kl=k2;
254  jh=j1;kh=k1;
255  }
256 
257  m = (kh -kl)/(Scalar)(jh - jl);
258  kt = kl;
259

261  memset(segments,0,4*(3+(jh-jl))*sizeof(int));
262 
263  /* make sure the first point is in the array */
264  segments[0]=jl;
265  segments[1]=kl;
266 
267  /* get all the horizontal segments. */
268  for(j=jl;j<jh;j++) {
269  y = m * (j - jl + 0.5) + kt;
270  k1s = (int) (y + 0.5);
271  k2s = k1s;
272  j1s = j;
273  j2s = j+1;
274  segments[4*segcount+2]=j1s;
275  segments[4*segcount+3]=k1s;
276  segments[4*segcount+4]=j2s;
277  segments[4*segcount+5]=k2s;
278  segcount++;
279  }
280

282  segments[4*segcount+2]=jh;
