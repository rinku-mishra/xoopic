104 Scalar Secondary2::number(Particle& p)
105 {
106  if (secondary <= 0) return 0;
107  Scalar theta, delta_ratio;
108  Vector3 v = p.get_u()/p.gamma();
109  theta = acos(-(v*bPtr->unitNormal())/v.magnitude());
110  Scalar Emax = energy_max0*(1 + 0.5*ks*theta*theta/M_PI);
111  Scalar w = (p.energy_eV_MKS() - threshold)/(Emax - threshold);
112  Scalar k = (w > 1) ? 0.25 : 0.56;
113  if (w < 3.6)
114  delta_ratio = pow(w*exp(1-w), k);
115  else
116  delta_ratio = 1.125/pow(w, Scalar(0.35));
117  return secondary*(1 + 0.5*ks*theta*theta/M_PI)*delta_ratio;
118 }
