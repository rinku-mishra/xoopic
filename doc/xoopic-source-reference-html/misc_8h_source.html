136 /* frand() returns values 0 through 1. */
137 /* From "Random number generators: good ones are hard to find", S. Park */
138 /* and K. Miller, Communications of ACM, October 1988, pp 1192-1201. */
139 /* This is from page 1195, and is to work on any system for which */
140 /* maxint is 2**31-1 or larger. Due earlier to Schrage, as cited by P&M.*/
141 /* */
142 /* Note: OK values for iseed are 1 through 2**31-2. Give it 0 or 2*31-1 */
143 /* and it will return the same values thereafter! */
144 /* */
145 /* C version 6/91, Bruce Langdon. */
146 /* */
147 /* Algorithm replaces seed by mod(a*seed,m). First represent */
148 /* seed = q*hi + lo. Then */
149 /* a*seed = a*q*hi + a*lo = (m - r)*hi + a*lo = (a*lo - r*hi) + m*hi, */
150 /* and new seed = a*lo - r*hi unless negative; if so, then add m. */
151 
152 extern int frandseed;
153 
154 inline Scalar frand()
155 {
156  int a = 16807, m = 2147483647, q = 127773, r = 2836;
157  int hi, lo;
158  Scalar fnumb;
159 // static long seed=31207321;
160 
161  // moved seed to frandseed
162  // fnumb = 2;
163  // the random number returned is 0=<fnumb<1
164  do {
165  hi = frandseed/q;
166  lo = frandseed - q*hi;
167  frandseed = a*lo - r*hi;
168  /* "seed" will always be a legal integer of 32 bits (including sign). */
169  // if(seed <= 0) seed = seed + m;
170  if(frandseed < 0) frandseed = frandseed + m;
171 #if (ScalarInt == 4)
172  fnumb = (frandseed & 0x7fffff80)/2147483647.0f;
173 #else
174  fnumb = frandseed/2147483647.0;
175 #endif
176  } while (fnumb-1.0 >= 0); // johnv: modified to handle single precision properly
177  return(fnumb);
178 }

187 // return a bit reversed number
188 inline Scalar revers(int num, int n)
189 {
190  Scalar power, rev;
191  int inum, iquot, irem;
192 
193  rev = 0.;
194  inum = num;
195  power = 1.;
196 
197  do
198  {
199  iquot = inum/n;
200  irem = inum - n*iquot;
201  power /= n;
202  rev += irem*power;
203  inum = iquot;
204  }
205  while (inum > 0);
206 
207  return (rev);
208 }
