76  Vector3 **Edl = region->getIntEdl();
77  Vector2 **X = region->getX();
78 
79  // Do the integral: I could do this more accurately...
80  // This is a first-order integral algorithm.
81  // Integrate Y*Y * Ey(Y) * Ey(Y)
82  // divided by Ey(Y)*Ey(Y)
83  for(k=0;k<size;k++) {
84  Scalar y = 0.5 * (X[j1][k].e2() + X[j1][k+1].e2());
85  Scalar Eydl = Edl[j1][k].e2();
86 // Weight1Sum += (y-yC) * sqr(Eydl);
87  WeightSum += sqr(y - yC) * sqr(Eydl);
88 // Weight2Sum += sqr(sqr(y-yC)) * sqr(Eydl);
89  NormSum += sqr(Eydl);
90  }
91 
92  past_measurements_y2Ey2->add(WeightSum,t);
93 // past_measurements_yEy2->add(Weight1Sum,t);
94 // past_measurements_y4Ey2->add(Weight2Sum,t);
95  past_measurements_Ey2->add(NormSum,t);
96 
97  // Average the past measurements and put it into history
98  NormSum = 0;
99  WeightSum=0;
100  if( (hist_num = *(past_measurements_Ey2->get_hi())) > 0) {
101  for(k=0;k<hist_num;k++) {
102  NormSum += past_measurements_Ey2->get_data()[k];
103  WeightSum += past_measurements_y2Ey2->get_data()[k];
104  }
105  }
106 
107  if(NormSum > 0) {
108  R = sqrt(WeightSum / NormSum);
109  }
110 
111  history->add(R,t);
