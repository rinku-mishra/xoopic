250 Scalar pulShpF(int pulshp, Scalar tdel, Scalar lengP, Scalar t)
251 {
252 
253  Scalar Amp, tp;
254  Scalar speedOfLight = 2.9979e08;
255  Scalar PI = 3.14159265358979323846;
256 
257  tp = lengP/speedOfLight;
258 
259  switch(pulshp)
260  {
261  // Trapezoidal pulse
262  case 0:
263 
264  if(t<tdel)
265  Amp = 0.0;
266  else if(t<tdel+tp*0.1)
267  Amp = (10.0/tp)*(t - tdel);
268  else if(t<tdel+0.9*tp)
269  Amp = 1.0;
270  else if(t<tdel+tp)
271  Amp = (10.0/tp)*(tdel+tp-t);
272  else
273  Amp = 0.0;
274 
275  break;
276 
277  // Gaussian pulse
278  case 1:
279 
280  if(t<tdel)
281  Amp = 0.0;
282  else if(t<tdel+tp)
283  Amp = exp(-4.0*(t-tdel-0.5*tp)*(t-tdel-0.5*tp)/(tp*tp));
284  else
285  Amp = 0.0;
286 
287  break;
288 
289  // Half-Sinusoidal pulse
290  case 2:
291 
292  if(t<tdel)
293  Amp = 0.0;
294  else if(t< tdel+tp)
295  Amp = sin(PI*(t-tdel)/tp);
296  else
297  Amp = 0.0;
298  break;
299 
300  // Wide Gaussian pulse
301  case 3:
302 
303  if(t<tdel-3*tp)
304  Amp = 0.0;
305  else if(t<tdel+3*tp)
306  // DLB (Oct. 3, 2003)
307  // Amp = exp(-4.0*(t-tdel-0.5*tp)*(t-tdel-0.5*tp)/(tp*tp));
308  Amp = exp( -(t-tdel)*(t-tdel) / (tp*tp) );
309  else
310  Amp = 0.0;
311 
312  break;
313 
314  // constant amplitude pulse
315  case 4:
316 
317  if(t<tdel)
318  Amp = 0.0;
319  else if(t<tdel+tp)
320  Amp = 1.0;
321  else
322  Amp = 0.0;
323 
324  break;
325 
326 
327  default:
328  Amp = 0.0;
329  break;
330 
331  }
332 
333  return Amp;
334 }
