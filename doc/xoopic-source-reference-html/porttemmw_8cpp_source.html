42  // px is the position in the plane wave

44  Scalar px = dx * (sRegion->getShiftNum() );
45 
46  // the E field at the edge of the leading edge cell
47  // (J-1) (should be unnecessary)
48  Scalar Eydl;
49 // Scalar Eydl = dy*A * cos(pwWavenumber * px + w * tE + pwPhase);

55  Eydl = dy * A * sin(pwWavenumber * px + w * tE + pwPhase);
56  for(k=0;k<K;k++) iEdl[J][k]=Vector3(0,Eydl,0);

70

73  Scalar m = ELECTRON_MASS;

75  if(! EF && ! NF) { //do the default thing.

89  x1 = px;

91  Scalar Scale = adv_eval->Evaluate(AnalyticEy);
92  if(NF) {
93  Scalar w2 = w*w;
94  N = adv_eval->Evaluate(AnalyticN);
95  Wp2 = N * e * e / (m * EPS0);
96  if(Wp2 >= w2)

98  else
99  nK = sqrt(w2 - Wp2) / SPEED_OF_LIGHT;
100  }
101  else nK = pwWavenumber;
102  Scalar BzdS = -dS*A*Scale*(nK / w)* sin( pwWavenumber *px + w *tB +pwPhase) ;

104  }
105  }
