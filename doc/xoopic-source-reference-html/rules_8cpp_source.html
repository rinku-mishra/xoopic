32 double BinaryOperator::evaluate() {
33  double result=0.;
34  op.strip(1); //strip any trailing whitespace.
35  if (op == ostring("==")) result = (value1 == value2);
36  if (op == ostring("<=")) result = (value1 <= value2);
37  if (op == ostring(">=")) result = (value1 >= value2);
38  if (op == ostring("<")) result = (value1 < value2);
39  if (op == ostring(">")) result = (value1 > value2);
40  if (op == ostring("+")) result = (value1 + value2);
41  if (op == ostring("-")) result = (value1 - value2);
42  if (op == ostring("*")) result = (value1 * value2);
43  if (op == ostring("/")) result = (value1/value2);
44  return result;
45 }
