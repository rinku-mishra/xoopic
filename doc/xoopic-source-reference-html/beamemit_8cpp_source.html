268 // computeLocationArray()
269 // precomputes a array of equally spaced points mapping the
270 // cumulative distribution to position along the emitter
271 // presently only works for orthogonal emitters. Note that
272 // geomFactor includes the radial nature if cylindrical.
273 // The integration uses the Trapezoidal Rule.
274 
275 void BeamEmitter::computeLocationArray()
276 {
277  Vector2 component(0,0);
278  if (alongx2()) component.set_e2(1);
279  else component.set_e1(1);
280  Scalar dx = ((p2 - p1)*component)/nIntervals;
281  xArray[0] = p1*component;
282  FArray[0] = 0;
283  // set up for initial pass thru loop
284  // The following line has a bug for the case of cylindrical
285  // geometry, when the emitter is aligned along the X1 (z)
286  // direction (perhaps an unlikely situation).
287  // Bruhwiler/Dimitrov 10/12/2000
288  // Scalar f1 = geomFactor(xArray[0])*get_xt_value(xArray[0],t);
289 
290  // The following three lines fix the bug noted above.
291  Scalar geometryFactor = 1.;
292  if ( alongx2() ) geometryFactor = geomFactor(xArray[0]);
293  Scalar f1 = geometryFactor * get_xt_value(xArray[0],t);
294 
295  int i;
296  for (i=1; i<=nIntervals; i++) {
297  Scalar f0 = f1; // i-1 value
298  xArray[i] = xArray[0] + i*dx;
299  // Here we fix the bug noted above in the same way.
300  // f1 = geomFactor(xArray[i])*get_xt_value(xArray[i], t);
301  geometryFactor = 1.;
302  if ( alongx2() ) geometryFactor = geomFactor(xArray[i]);
303  f1 = geometryFactor*get_xt_value(xArray[i], t);
304 
305  FArray[i] = FArray[i-1] + 0.5*dx*(f0 + f1); // trapezoidalintegration
306  }
307  FTotal = FArray[nIntervals];
308  // avoid a division by zero
309  if(FTotal <= FLT_MIN) { fAvg = 0; return;};
310  for (i=1; i<=nIntervals; i++) FArray[i] /= FTotal;
311  Scalar temp;
312  if (alongx2())
313  if (rweight) temp = 0.5*(rMax*rMax - rMin*rMin);
314  else temp = p2.e2() - p1.e2();
315  else temp = p2.e1() - p1.e1();
316  fAvg = FTotal/temp;
317 }

324 Vector2 BeamEmitter::computeLocation()
325 {
326  Scalar F = frand();
327  if (get_xtFlag() < 2) {
328  if (rweight && alongx2()) return Vector2(0.5*(p1.e1() + p2.e1()),
329  sqrt(rMin*rMin + (rMax*rMax - rMin*rMin)*frand()));

331  }
332  else {
333  int i;
334  for (i=0; i<nIntervals; i++) if (F < FArray[i]) break;
335  Scalar x = xArray[i-1] + (xArray[i] - xArray[i-1])*(F - FArray[i-1])/(FArray[i] - FArray[i-1]);
336  if (alongx1()) return Vector2(x, p1.e2());
337  else return Vector2(p1.e1(), x);
